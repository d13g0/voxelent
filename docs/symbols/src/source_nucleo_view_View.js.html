 17 /**
 18 * Each HTML5 canvas is assigned a view object (vxlView) in Voxelent's Nucleo.
 19 * 
 20 * A vxlView provides the code to handle cameras, interaction and rendering capabilities, plus actor handling on the 
 21 * HTML5 canvas that otherwise would need to be written over and over for each application
 22 * if you were writing a WebGL app from scratch.
 23 * Luckily this is not the case. You have the awesome vxlView who takes care of all these little details for you.
 24 * @class the class that manages what it is rendered the HTML5 canvas on the web page 
 25 * @constructor
 26 * @param {String} canvasID id of the canvas in the DOM tree. That's all we need to setup a vxlView for you
 27 * @param {vxlScene} scene if this view is sharing a scene, this parameter corresponds to the scene being shared.
 28 * @param {Boolean} handleLayout if true or absent, the canvas will be resized dynamically
 29 * @author Diego Cantor
 30 */
 31 
 32 function vxlView(canvasID, scene, handleLayout){
 33 
 34 	//this.id = 0; //@TODO: Who handles this? Maybe one Scene has several Views?
 35 	
 36 	this.canvas = document.getElementById(canvasID);
 37 	
 38 	if (this.canvas == null){
 39 		alert('View: the canvas ' + canvasID+ ' does not exist.');
 40 		throw('View: the canvas ' + canvasID+ ' does not exist.');
 41 	}
 42 	
 43 	this.name = canvasID;
 44 	
 45     //View dimensions
 46 	this.width = this.canvas.width;
 47 	this.height = this.canvas.height;
 48 	
 49 	//Clear depth and background color
 50 	this.clearDepth = 1.0;
 51 	this.backgroundColor = vxl.def.view.background.slice(0);
 52 	
 53 	//Drag and drop
 54 	this.dragndrop = false;
 55 
 56 	//Create Renderer
 57 	this.renderer = new vxlRenderer(this);
 58 	this.setBackgroundColor(this.backgroundColor);
 59 	this.setClearDepth(this.clearDepth);
 60 	
 61 	//Create Camera Manager
 62 	this.cameraman = new vxlCameraManager(this);
 63 	
 64 	//Create View Interactor
 65 	this.interactor = new vxlTrackerInteractor();
 66 	
 67 	this.interactor.connectView(this);
 68 	
 69 	//Create Scene
 70 	if (scene != null && scene instanceof vxlScene){
 71 		this.scene = scene;
 72 	}
 73 	else if (vxl.c.scene != undefined){
 74 		this.scene = vxl.c.scene;
 75 	}
 76 	else{
 77 		this.scene = new vxlScene();
 78 	
 79 	}
 80 	//Add this view to the scene;
 81 	this.scene.views.push(this);
 82 	
 83 	if (vxl.c.view == undefined){
 84 		vxl.c.view = this;
 85 	}
 86 	
 87 	vxl.go.views.push(this);
 88 	
 89 	if (handleLayout == undefined){
 90 	    handleLayout = true;
 91 	}
 92 	
 93 	this.setAutoResize(handleLayout);
 94 	this._fullscreenFlag = true;
 95 	
 96 	this.UID = vxl.util.generateUID();
 97 
 98 
 99     var ntf = vxl.go.notifier;
100     var e   = vxl.events;
101     ntf.publish(e.VIEW_NEW, this);
102     ntf.subscribe(e.DEFAULT_LUT_LOADED,this);
103     ntf.fire(e.VIEW_NEW, this);
104 
105 	vxl.go.console('View: the view '+ this.name+' has been created');
106 };

109 /**
110  * Handles the events to which a view is subscribed in Voxelent's Nucleo
111  * @param {String} event the name of the event
112  * @param {Object} source the origin of the event. Useful to do callbacks if necessary
113  */
114 vxlView.prototype.handleEvent = function(event, source){
115 	vxl.go.console('vxlView ['+this.name+']: receiving event '+event);
116 	if (event == vxl.events.DEFAULT_LUT_LOADED){
117 		this.scene.setLookupTable(vxl.def.lut.main);
118 	}
119 };

129 /**
130  * It creates a div around the canvas to handle resizing appropriately
131  * @private 
132  */
133 vxlView.prototype._wrapDiv = function(){
134     
135     $(this.canvas).css({
136         position: 'absolute',
137         height: 'auto',
138         bottom: 0,
139         top: 0,
140         left: 0,
141         right: 0,
142         margin: "5px"
143     });
144     
145     $(this.canvas).wrap('<div id='+this.name+'-wrapper />');
146     $('#'+this.name+'-wrapper').css({
147        width:'100%',
148        height:'100%' 
149     });
150     
151     this.canvas.focus();
152 }

154 /**
155  * Resizes the canvas so it fits its parent node in the DOM tree
156  */
157 vxlView.prototype.resize = function(){
158     var parent = this.canvas.parentNode;
159     this.width = $(parent).width()-5;       
160     this.height = $(parent).height()-5;
161     
162     if (this.width > window.innerWidth - 5){
163     	this.width = window.innerWidth -5;
164     }
165     
166     if (this.height > window.innerHeight - 5){
167     	this.height = window.innerHeight - 5;
168     }
169     
170     $(this.canvas).attr('width', this.width);
171     $(this.canvas).attr('height', this.height);
172 }
173 
174 /**
175  * Enables automatic resizing of the view when the browser window is resized
176  * @param flag enbles automatic resizing if true, disables it if false
177  */
178 vxlView.prototype.setAutoResize = function(flag){
179     
180     if (flag){
181          this._wrapDiv();
182          this.resize(); //first time
183         $(window).resize((function(self){return function(){self.resize();}})(this));
184     }
185     else{
186         $(window).unbind('resize');
187     }
188 };

191 /**
192  * Run non standard code (mozilla, webkit) 
193  *@private 
194  */
195 vxlView.prototype._runPrefixMethod = function(obj,method){
196     
197     var pfx = ["webkit", "moz", "ms", "o", ""];
198     var p = 0, m, t;
199 
200     while (p < pfx.length && !obj[m]) {
201         m = method;
202         if (pfx[p] == "") {
203             m = m.substr(0,1).toLowerCase() + m.substr(1);
204         }
205         m = pfx[p] + m;
206         t = typeof obj[m];
207         if (t != "undefined") {
208             pfx = [pfx[p]];
209             return (t == "function" ? obj[m]() : obj[m]);
210         }
211         p++;
212     }
213 };

215 /**
216  * Handles HTML5 Full screen for this view 
217  * @param {Boolean} flag set flag to true if fullscreen is wanted. Set it to false to exit fullscreen.
218  */
219 vxlView.prototype.fullscreen = function(flag){
220     if (!this._fullscreenFlag) return; //ignore request if fullscreen is not active
221     if (flag == true){
222         this._runPrefixMethod(this.canvas, "RequestFullScreen");
223     }
224     else{
225         if (this._runPrefixMethod(document,"FullScreen") || 
226             this._runPrefixMethod(document,"isFullScreen")){
227             this._runPrefixMethod(document,"CancelFullScreen");
228         }    
229     }
230 
231 };

272 /**
273  * Sets the view's background color
274  * @param {Number, Array, vec3} r it can be the red component, a 3-dimensional Array or a vec3 (glMatrix)
275  * @param {Number} g if r is a number, then this parameter corresponds to the green component
276  * @param {Number} b if r is a number, then this parameter corresponds to the blue component
277  
278  * @see vxlRenderer#clearColor
279  */
280 vxlView.prototype.setBackgroundColor = function(r,g,b){
281 	this.backgroundColor = vxl.util.createColor(r,g,b); 
282 	this.renderer.clearColor(this.backgroundColor);
283 };
