 17 /**
 18 * Each HTML5 canvas is assigned a view object (vxlView) in Voxelent's Nucleo.
 19 * 
 20 * A vxlView provides the code to handle cameras, interaction and rendering capabilities, plus actor handling on the 
 21 * HTML5 canvas that otherwise would need to be written over and over for each application
 22 * if you were writing a WebGL app from scratch.
 23 * Luckily this is not the case. You have the awesome vxlView who takes care of all these little details for you.
 24 * @class
 25 * @constructor
 26 * @param {String} canvasID id of the canvas in the DOM tree. That's all we need to setup a vxlView for you
 27 * @param {vxlScene} scene if this view is sharing a scene, this parameter corresponds to the scene being shared.
 28 * @author Diego Cantor
 29 */
 30 
 31 function vxlView(canvasID, scene){
 32 	//View Identification
 33 	//this.id = 0; //@TODO: Who handles this? Maybe one Scene has several Views?
 34 	this.name = canvasID;
 35 	this.canvas = document.getElementById(this.name);
 36 	if (this.canvas == null){
 37 		alert('View: the canvas ' + canvasID+ ' does not exist.');
 38 		throw('View: the canvas ' + canvasID+ ' does not exist.');
 39 	}
 40 	
 41 
 42 	this.width = this.canvas.width;
 43 	this.height = this.canvas.height;
 44 	
 45 	this.clearDepth = 1.0;
 46 	this.backgroundColor = vxl.def.view.background.slice(0);
 47 
 48 	//Create Renderer
 49 	this.renderer = new vxlRenderer(this);
 50 	this.setBackgroundColor(this.backgroundColor);
 51 	this.setClearDepth(this.clearDepth);
 52 	
 53 	//Create Camera Manager
 54 	this.cameraman = new vxlCameraManager(this);
 55 	
 56 	//Create View Interactor
 57 	this.interactor = new vxlTrackerInteractor(this);
 58 	
 59 	//Create Scene
 60 	if (scene != null && scene instanceof vxlScene){
 61 		this.scene = scene;
 62 	}
 63 	else if (vxl.c.scene != undefined){
 64 		this.scene = vxl.c.scene;
 65 	}
 66 	else{
 67 		this.scene = new vxlScene();
 68 	
 69 	}
 70 	//Add this view to the scene;
 71 	this.scene.views.push(this);
 72 	
 73 	
 74 	//Namespace access updates
 75 	
 76 
 77 	vxl.go.notifier.addTarget(vxl.events.DEFAULT_LUT_LOADED,this);     //Register events to listen to
 78 	
 79 	if (vxl.c.view == undefined){
 80 		vxl.c.view = this;
 81 	}
 82 	
 83 	vxl.go.views.push(this);
 84 	this.setAutoResize(true);
 85 
 86 	vxl.go.console('View: the view '+ this.name+' has been created');
 87 };
 88 
 89 
 90 /**
 91  * Handles the events to which a view is subscribed in Voxelent's Nucleo
 92  * @param {String} event the name of the event
 93  * @param {Object} source the origin of the event. Useful to do callbacks if necessary
 94  */
 95 vxlView.prototype.handleEvent = function(event, source){
 96 	vxl.go.console('vxlView ['+this.name+']: receiving event '+event);
 97 	if (event == vxl.events.DEFAULT_LUT_LOADED){
 98 		this.scene.setLookupTable(vxl.def.lut.main);
 99 	}
100 };

103 /**
104  * Clears the scene. Delegates this task to the renderer.
105  */
106 vxlView.prototype.clear = function(){
107 	this.renderer.clear();
108 };
109 
110 /**
111  * Resizes the canvas so it fits its parent node in the DOM tree
112  */
113 vxlView.prototype.resize = function(){
114     var parent = this.canvas.parentNode;
115     this.width = $(parent).width()-5;       //@TODO: Review this -5 business here.... March 12/2012
116     this.height = $(parent).height()-5;
117     $(this.canvas).attr('width', this.width);
118     $(this.canvas).attr('height', this.height);
119 }
120 
121 /**
122  * Enables automatic resizing of the view when the browser window is resized
123  * @param flag enbles automatic resizing if true, disables it if false
124  */
125 vxlView.prototype.setAutoResize = function(flag){
126     if (flag){
127          this.resize(); //first time
128         $(window).resize((function(self){return function(){self.resize();}})(this));
129     }
130     else{
131         $(window).unbind('resize');
132     }
133 }
134 
135 /**
136  * Starts the view
137  */
138 vxlView.prototype.start = function(){
139 	this.renderer.start();
140 	this.refresh();
141 };
142 
143 /**
144  * Stops the view
145  */
146 vxlView.prototype.stop = function(){
147 	this.renderer.stop();
148 };
149 
150 /**
151  * Resets the view
152  */
153 vxlView.prototype.reset = function(){
154 	this.stop();
155 	this.scene.reset();
156 	this.cameraman.reset();
157 	this.start();
158 };
159 
160 /**
161  * Sets the background color. Delegated to the renderer
162  * @param {Array} v the new color given as an array of three numbers
163  * @see vxlRenderer#clearColor
164  */
165 vxlView.prototype.setBackgroundColor = function(v){
166 	this.backgroundColor = v.slice(0);
167 	this.renderer.clearColor(this.backgroundColor);
168 };
169 
170 
171 /**
172  * Sets the clear depth
173  * @param {Number} d the new depth
174  * @see vxlRenderer#clearDepth
175  */
176 vxlView.prototype.setClearDepth = function(d){
177 	this.renderer.clearDepth(d);
178 };
179 
180 /**
181  * Refresh the view by invoking the rendering method on the renderer
182  * @see vxlRenderer#render
183  */
184 vxlView.prototype.refresh = function(){
185 	this.renderer.render();
186 };
187 
188 /**
189  * Uses the interactor passed as parameter to handle user gestures
190  * @param {vxlViewInteractor} interactor an instance of a vxlViewInteractor
191  * 
192  */
193  vxlView.prototype.setInteractor = function(i){
194     this.interactor = i;
195     this.interactor.connectView(this);
196 };
