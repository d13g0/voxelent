 18 vxlBasicStrategy.prototype = new vxlRenderStrategy(undefined)
 19 vxlBasicStrategy.prototype.constructor = vxlBasicStrategy;

 21 /**
 22  * @constructor
 23  * @class
 24  * Implements a basic rendering strategy that works with the following programs:
 25  * 
 26  * vxl.def.glsl.lambert
 27  * vxl.def.glsl.phong
 28  * 
 29  * A vxlBasicStrategy object is selected by default as the strategy that an actor
 30  * will use for rendering.
 31  * 
 32  * A rendering strategy allows decoupling rendering specific code (i.e. code that access and 
 33  * 	communicates with the program) from the actor.
 34  * 
 35  */
 36 function vxlBasicStrategy(renderer){
 37 	vxlRenderStrategy.call(this,renderer);
 38 }
 39 
 40 /**
 41  * Implements basic allocation of memory. Creates the WebGL buffers for the actor
 42  * @param{vxlActor} actor the actor to allocate memory for
 43   * @returns an object that contains the allocated WebGL buffers
 44  */
 45 vxlBasicStrategy.prototype.allocate = function(actor){
 46    
 47     vxl.go.console('Actor: Allocating actor '+actor.name+' for view '+ this.renderer.view.name);
 48    	
 49 	var gl = this.renderer.gl;
 50 	var model = actor.model;
 51     var buffers = {};
 52 	
 53 	//Vertex Buffer
 54 	buffers.vertex = gl.createBuffer();
 55 	gl.bindBuffer(gl.ARRAY_BUFFER, buffers.vertex);
 56 	gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(model.vertices), gl.STATIC_DRAW);
 57 	
 58 	//Index Buffer
 59 	if (model.indices != undefined){
 60 		buffers.index = gl.createBuffer();
 61 		gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, buffers.index);
 62 		gl.bufferData(gl.ELEMENT_ARRAY_BUFFER, new Uint16Array(model.indices), gl.STATIC_DRAW);
 63 	}
 64 	
 65 	//Normals Buffer
 66 	if (model.normals){
 67 		buffers.normal = gl.createBuffer();
 68 		gl.bindBuffer(gl.ARRAY_BUFFER, buffers.normal);
 69 		gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(model.normals), gl.STATIC_DRAW);
 70 	}
 71 	
 72 	//Color Buffer for scalars
 73 	if (model.scalars != undefined || model.colors != undefined){
 74 		buffers.color = gl.createBuffer(); //we don't BIND values or use the buffers until the lut is loaded and available
 75 	}
 76 	
 77 	//Wireframe Buffer 
 78 	if (model.wireframe != undefined){
 79 		buffers.wireframe = gl.createBuffer();
 80 		gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, buffers.wireframe);
 81 		gl.bufferData(gl.ELEMENT_ARRAY_BUFFER, new Uint16Array(model.wireframe), gl.STATIC_DRAW);
 82 	}
 83 	
 84 	//Cleaning up
 85 	gl.bindBuffer(gl.ARRAY_BUFFER, null);
 86 	gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, null);
 87 	
 88 	return buffers;
 89 };

 91 /**
 92  * Passes the matrices to the shading program
 93  * @param {vxlActor} the actor 
 94  * we will update each Model-View matrix of each renderer according to
 95  * the actor position,scale and rotation.
 96  */
 97 vxlBasicStrategy.prototype.applyActorTransform = function(actor){
 98     
 99     var r		= this.renderer;
100     var trx 	= r.transforms
101     var	prg 	= r.prg;
102     var glsl 	= vxl.def.glsl;
103     
104     trx.calculateModelView();
105     
106     trx.push();
107         mat4.scale      (trx.mvMatrix, actor.scale);
108 		mat4.translate	(trx.mvMatrix, actor.position);
109 		//@TODO: IMPLEMENT ACTOR ROTATIONS
110 	    prg.setUniform(glsl.MODEL_VIEW_MATRIX,	r.transforms.mvMatrix);
111     trx.pop();
112     
113     trx.calculateNormal(); 
114     
115     prg.setUniform(glsl.PERSPECTIVE_MATRIX, 	r.transforms.pMatrix);
116     prg.setUniform(glsl.NORMAL_MATRIX, 			r.transforms.nMatrix);
117     
118  };

121 vxlBasicStrategy.prototype.render = function(actor){
122 
123 	var model 	= actor.model;
124 	var idx 	= actor._renderers.indexOf(this.renderer);
125     var buffers = actor._gl_buffers[idx]; 
126     var r  		= this.renderer;
127 	var gl 		= r.gl;
128 	var prg 	= r.prg;
129 	var trx 	= r.transforms;
130 	var glsl    = vxl.def.glsl;
131 	
132 	this.applyActorTransform(actor);
133 	
134 	var diffuse = [actor.color[0], actor.color[1], actor.color[2],1.0];
135 	
136 	if (actor.opacity < 1.0){
137 		gl.disable(gl.DEPTH_TEST);
138 		gl.enable(gl.BLEND);
139 		diffuse[3] = actor.opacity;
140 	}
141 	else {
142 		gl.enable(gl.DEPTH_TEST);
143 		gl.disable(gl.BLEND);
144 		diffuse[3] = 1.0;
145 	}
146 	
147 	prg.setUniform("uMaterialDiffuse",diffuse);
148 	prg.setUniform("uUseVertexColors", false);
149     
150     prg.disableAttribute(glsl.COLOR_ATTRIBUTE);
151 	prg.disableAttribute(glsl.NORMAL_ATTRIBUTE);
152 	prg.enableAttribute(glsl.VERTEX_ATTRIBUTE);
153 	
154 	try{
155 		
156         gl.bindBuffer(gl.ARRAY_BUFFER, buffers.vertex);
157         gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(model.vertices.slice(0)), gl.STATIC_DRAW);
158 		prg.setAttributePointer(glsl.VERTEX_ATTRIBUTE, 3, gl.FLOAT, false, 0, 0);
159 	}
160     catch(err){
161         alert('There was a problem while rendering the actor ['+actor.name+']. The problem happened while handling the vertex buffer. Error =' +err.description);
162 		throw('There was a problem while rendering the actor ['+actor.name+']. The problem happened while handling the vertex buffer. Error =' +err.description);
163     }
164     	
165 	if (actor.colors){	
166 		try{
167 			prg.setUniform("uUseVertexColors", true);
168 			gl.bindBuffer(gl.ARRAY_BUFFER, buffers.color);
169 			gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(actor.colors.slice(0)), gl.STATIC_DRAW);
170 			
171 			prg.enableAttribute(glsl.COLOR_ATTRIBUTE);
172 			prg.setAttributePointer(glsl.COLOR_ATTRIBUTE, 3, gl.FLOAT, false, 0, 0);
173 		}
174 		catch(err){
175         	alert('There was a problem while rendering the actor ['+actor.name+']. The problem happened while handling the color buffer. Error =' +err.description);
176 			throw('There was a problem while rendering the actor ['+actor.name+']. The problem happened while handling the color buffer. Error =' +err.description);
177    		}
178     }
179     
180     if(model.normals){
181 	    try{
182 			gl.bindBuffer(gl.ARRAY_BUFFER, buffers.normal);
183 			gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(model.normals.slice(0)), gl.STATIC_DRAW);
184 			
185 			prg.enableAttribute(glsl.NORMAL_ATTRIBUTE);
186 			prg.setAttributePointer(glsl.NORMAL_ATTRIBUTE,3,gl.FLOAT, false, 0,0);
187 		}
188 	    catch(err){
189 	        alert('There was a problem while rendering the actor ['+actor.name+']. The problem happened while handling the normal buffer. Error =' +err.description);
190 			throw('There was a problem while rendering the actor ['+actor.name+']. The problem happened while handling the normal buffer. Error =' +err.description);
191 	    }
192 	}
193     prg.setUniform("uUseShading",actor.shading);
194     
195     try{
196 		if (actor.mode == vxl.def.actor.mode.SOLID){
197 			prg.enableAttribute(glsl.NORMAL_ATTRIBUTE);
198 			gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, buffers.index);
199 			gl.drawElements(gl.TRIANGLES, model.indices.length, gl.UNSIGNED_SHORT,0);
200 		}
201 		else if (actor.mode == vxl.def.actor.mode.WIREFRAME){
202 			if (actor.name == 'floor'){
203 			     prg.disableAttribute(glsl.NORMAL_ATTRIBUTE);
204 			}
205 			gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, buffers.wireframe);
206 			gl.bufferData(gl.ELEMENT_ARRAY_BUFFER, new Uint16Array(model.wireframe), gl.STATIC_DRAW);
207 			gl.drawElements(gl.LINES, model.wireframe.length, gl.UNSIGNED_SHORT,0);
208 		}
209 		else if (actor.mode == vxl.def.actor.mode.POINTS){
210 			prg.setUniform("uUseShading", true);
211 			prg.enableAttribute(glsl.NORMAL_ATTRIBUTE);
212 			gl.drawArrays(gl.POINTS,0, this.model.vertices.length/3);
213 		}
214 		else if (actor.mode == vxl.def.actor.mode.LINES){
215 			prg.setUniform("uUseShading", false);
216 			prg.disableAttribute(glsl.NORMAL_ATTRIBUTE);
217 			gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, buffers.index);
218 			gl.drawElements(gl.LINES, actor.model.indices.length, gl.UNSIGNED_SHORT,0);
219 		}
220 		else{
221             alert('There was a problem while rendering the actor ['+actor.name+']. The visualization mode: '+this.mode+' is not valid.');
222 			throw('There was a problem while rendering the actor ['+actor.name+']. The visualization mode: '+this.mode+' is not valid.');
223 			
224 		}
225 		//Cleaning up
226 		 gl.bindBuffer(gl.ARRAY_BUFFER, null);
227 	     gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, null);
228     }
229 	catch(err){
230 		alert('Error rendering actor ['+actor.name+']. Error =' +err.description);
231 		throw('Error rendering actor ['+actor.name+']. Error =' +err.description);
232 	}
233 };
