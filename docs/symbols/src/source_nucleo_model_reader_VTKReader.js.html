 17 /**
 18  * Reads a VTK file and creates the respective JSON(s) file(s).
 19  * @class
 20  * @constructor 
 21  */
 22 function vxlVTKReader(scene){
 23 
 24     this.scene      = scene; //the scene that will be updated
 25     this.ARRAY_SIZE = 65536*3;
 26     this.vertices   = [];
 27     this.indices    = [];
 28     this.normals    = [];
 29     this.scalars    = [];
 30     this.colors     = [];
 31     
 32 
 33     this.mode = "SOLID";
 34 
 35     this.tags = {
 36         NOWHERE             : 0,    
 37         POINTS              : 1,
 38         LINES               : 2,
 39         POLYGONS            : 3,
 40         POINT_DATA          : 4,
 41         NORMALS             : 5,
 42         CELL_DATA           : 6,
 43         TEXTURE_COORDINATES : 7,
 44         SCALARS             : 8,
 45         LOOKUP_TABLE        : 9,
 46         COLOR_SCALARS       : 10
 47     };
 48     
 49     this.parts = [];
 50     
 51     vxl.go.notifier.publish(vxl.events.READER_DONE, this);
 52 };
 53 
 54 /**
 55  * Verifies if the HTML5 API is available. 
 56  */
 57 vxlVTKReader.prototype.isSupported = function(){
 58     return (window.File && window.FileReader && window.FileList && window.Blob);
 59 };

 62 /**
 63  * @param {File} file an HTML5 File object
 64  */
 65 vxlVTKReader.prototype.read = function(file){
 66     var vtkReader = this;
 67     var filename = file.name;
 68     
 69     if (filename.length - 4 !== filename.indexOf('.vtk')){
 70         throw 'vxlVTKReader.read: the file '+file+' is not a VTK file';
 71     }
 72     
 73     var reader = new FileReader();
 74      
 75     reader.onload = function(event){
 76         document.body.style.cursor = 'default';
 77         var contents = event.target.result.trim();
 78         var lines = contents.split(/\r\n|\r|\n/);
 79         vtkReader._parse(lines);  
 80         vtkReader._processIndexBlocks(filename);
 81         vxl.go.notifier.fire(vxl.events.READER_DONE, vtkReader);
 82     };
 83     
 84     document.body.style.cursor = 'wait';
 85     reader.readAsText(file);
 86 };

 88 /**
 89  * Reads the file line by line and populates the respective arrays
 90  */
 91 vxlVTKReader.prototype._parse = function(lines){
 92    this.outputfile = '';
 93    this.numBlocks = 0;
 94    this.location = 'NOWHERE';
 95    var linenumber = 0;
 96    for(var linenumber=0; linenumber<lines.length; linenumber++)
 97    {
 98     
 99        try
100             {
101                 if (lines[linenumber].indexOf('POINTS') == 0)
102                 {
103                     console.log(lines[linenumber]);
104                     this.location = this.tags.POINTS;
105                     continue;
106                 }
107                 else if (lines[linenumber].indexOf('LINES') == 0)
108                 {
109                     console.log(lines[linenumber]);
110                     this.location = this.tags.LINES;
111                     this.mode = "LINES";
112                     continue;
113                 }
114                 else if (lines[linenumber].indexOf('POLYGONS')==0)
115                 {
116                     console.log(lines[linenumber]);
117                     this.location = this.tags.POLYGONS;
118                     continue;
119                 }
120                 else if (lines[linenumber].indexOf('POINT_DATA')==0)
121                 {                   
122                     this.location = this.tags.POINT_DATA;
123                     continue;
124                 }
125                 else if (lines[linenumber].indexOf('NORMALS')==0)
126                 {      
127                     console.log(lines[linenumber]);             
128                     this.location = this.tags.NORMALS;
129                     continue;
130                 }
131                 else if (lines[linenumber].indexOf('CELL_DATA')==0)
132                 {      
133                     console.log(lines[linenumber]);             
134                     this.location = this.tags.CELL_DATA;
135                     continue;
136                 }
137                 else if (lines[linenumber].indexOf('TEXTURE_COORDINATES')==0)
138                 {      
139                     console.log(lines[linenumber]);             
140                     this.location = this.tags.TEXTURE_COORDINATES;
141                     continue;
142                 }
143                 else if (lines[linenumber].indexOf('SCALARS')==0)
144                 {      
145                     console.log(lines[linenumber]);             
146                     this.location = this.tags.SCALARS;
147                     continue;
148                 }
149                 else if (lines[linenumber].indexOf('LOOKUP_TABLE')==0)
150                 {      
151                     console.log(lines[linenumber]);             
152                     this.location = this.tags.LOOKUP_TABLE;
153                     continue;
154                 }   
155                 else if (lines[linenumber].indexOf('COLOR_SCALARS')==0)
156                 {      
157                     console.log(lines[linenumber]);             
158                     this.location = this.tags.COLOR_SCALARS;
159                     continue;
160                 }           
161                 // -------------------
162                 else if(this.location == this.tags.POINTS)
163                 {
164                     var v = lines[linenumber].trim().split(' ');
165                     if (v == "") continue;
166                     for (var i=0; i<v.length; i++)
167                     {
168                          this.vertices.push(parseFloat(v[i]));
169                     }
170                 }
171                 else if(this.location == this.tags.LINES)
172                 {
173                     var tt = lines[linenumber].trim().split(' ');
174                     if (tt == "") continue;
175                     if(tt.length>0 && tt[0] == '2')
176                     {
177                         this.indices.push(parseInt(tt[1]));
178                         this.indices.push(parseInt(tt[2]));
179                     }
180                 }
181                else if(this.location == this.tags.POLYGONS) //they have to be triangles
182                 {
183                     var tt = lines[linenumber].trim().split(' ');
184                     if (tt=="") continue; 
185                     if(tt.length>0 && tt[0] != '3')
186                     {
187                         throw "Not triangles here";
188                     }
189                     for(var i= 1; i<tt.length; i++)
190                     {
191                         this.indices.push(parseInt(tt[i]));
192                     }
193                 }
194                 else if(this.location == this.tags.LOOKUP_TABLE)
195                 {
196                     if(lines[linenumber].indexOf('LOOKUP_TABLE')==0)
197                         continue;
198                     else
199                     {
200                         var pd = lines[linenumber].trim().split(' ');
201                         if (pd=="") continue;
202                         for(var i=0; i<pd.length; i++)
203                         {
204                             this.scalars.push(parseFloat(pd[i]));
205                         }
206                    }
207                 }
208                 else if(this.location == this.tags.COLOR_SCALARS)
209                 {
210                     var n = lines[linenumber].trim().split(' ');
211                     if (n=="") continue;
212                     for(var i=0; i<n.length; i++)
213                     {
214                         this.colors.push(parseFloat(n[i]));
215                     }
216                 }
217                 else if(this.location == this.tags.NORMALS)
218                 {
219                     var n = lines[linenumber].trim().split(' ');
220                     if (n=="") continue;
221                     for(var i=0; i<n.length; i++)
222                     {
223                         this.normals.push(parseFloat(n[i]));
224                     }
225                 }
226            } // end try
227         catch(err)
228             {
229             console.log('Error while processing line '+ linenumber.toString());
230             //console.log(lines) // what is this for??
231             throw err;
232             }
233     }// end foor loop
234 };

236 /**
237  * Divides the calculated indices into blocks
238  * @private
239  */
240 vxlVTKReader.prototype._processIndexBlocks = function(filename){
241    this.numBlocks = 0;
242    var v_count =  this.vertices.length/3;
243    var n_count =  this.normals.length/3;
244    var ii_count = this.indices.length;
245    var i_count =  ii_count/3;
246    var c_count =  this.colors.length/3;
247    var pd_count = this.scalars.length;
248  
249    console.log('vertices:\t' + v_count.toString() +'\nnormals:\t' + n_count.toString() + '\nindices:\t' + ii_count.toString() +'\ntriangles:\t' + i_count.toString() +
250              '\nscalars:\t' + pd_count.toString() + '\ncolors:\t'+ c_count.toString()+'\n');
251  
252     this.numBlocks = parseInt(ii_count/this.ARRAY_SIZE);
253     if (ii_count % this.ARRAY_SIZE != 0)
254         this.numBlocks  = this.numBlocks + 1;
255     console.log( 'Number of Blocks: ' + this.numBlocks.toString());
256     
257     for(var i=0; i<this.numBlocks; i++){
258         this._processBlock(i, filename);
259     }
260 };

262 /**
263  * Process the selected block of indices
264  * @private
265  */
266 vxlVTKReader.prototype._processBlock = function(blockID, filename){
267     var pver = this.vertices;
268     var pidx = this.indices;
269     var psc = this.scalars;
270     var pcol = this.colors;
271     
272     var fid = (blockID + 1).toString();
273     var blockname = "";
274     
275     if (this.numBlocks == 1)
276         blockname = filename ;
277     else
278         blockname = filename + '_' + fid;
279         
280     //  _weaveBlock(blockID);
281     this._writeJSON(filename, pver, pidx, psc, pcol);
282     console.log('Block ['+ fid +'] processed,  output: '+ blockname);
283 };

351 /**
352  * Creates the JSON object
353  * @private 
354  */
355 vxlVTKReader.prototype._writeJSON = function(fname,ver,ind,pod,clr){
356      var jsonPart = new Object();
357      jsonPart["name"]        = fname;
358      jsonPart["vertices"]    = ver.slice(0);
359      jsonPart["indices"]     = ind.slice(0);
360      
361      if (pod.length>0){
362         jsonPart["scalars"]  = pod.slice(0);
363      }
364      
365      if (clr.length >0){
366         jsonPart["colors"]   = clr.slice(0);
367      }
368      
369      jsonPart["mode"]        = this.mode;
370      
371      this.parts.push(jsonPart);
372 };
