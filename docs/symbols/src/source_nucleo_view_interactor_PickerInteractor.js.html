 17 vxlPickerInteractor.prototype = new vxlTrackerInteractor();
 18 vxlPickerInteractor.prototype.constructor = vxlPickerInteractor;
 19 /**
 20  * @class 
 21  * Interactor that implements a picking mechanism. Still in development.
 22  * @constructor   
 23  * @param {Object} view the view this interactor will observe
 24  * @param {Object} camera the camera this interactor will master
 25  * @author Diego Cantor
 26  */
 27 function vxlPickerInteractor(view, camera){
 28 	vxlTrackerInteractor.call(this, view, camera);
 29 	this.plist 					= [];
 30 	this.texture 				= null;
 31 	this.framebuffer 			= null;
 32 	this.renderbuffer 			= null;
 33 	this.hitPropertyCallback 	= null;
 34 	this.processHitsCallback 	= null;
 35 	this.addHitCallback 		= null;
 36 	this.removeHitCallback 		= null;
 37 	this.moveCallback 			= null;
 38 	this.picking 				= false;
 39     this.connectCamera(camera);
 40 };
 41 
 42 vxlPickerInteractor.prototype.connectCamera = function(camera){
 43 	vxlViewInteractor.prototype.connectCamera.apply(this, camera);
 44 	if (this.camera){
 45 		this.configure();
 46 	}
 47 };

 49 vxlPickerInteractor.prototype.configure = function(){
 50 	
 51 	var gl = this.camera.view.renderer.gl;
 52 	
 53 	var width = 512*4; //@TODO get size from this.camera.view.canvas
 54 	var height = 512*4;
 55 	
 56 	//1. Init Picking Texture
 57 	this.texture = gl.createTexture();
 58 	gl.bindTexture(gl.TEXTURE_2D, this.texture);
 59 	gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MAG_FILTER, gl.LINEAR);
 60 	gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, gl.LINEAR_MIPMAP_NEAREST);
 61     gl.generateMipmap(gl.TEXTURE_2D);
 62     gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA, width, height, 0, gl.RGBA, gl.UNSIGNED_BYTE, null);
 63 	
 64 	//2. Init Render Buffer
 65 	this.renderbuffer = gl.createRenderbuffer();
 66     gl.bindRenderbuffer(gl.RENDERBUFFER, this.renderbuffer);
 67     gl.renderbufferStorage(gl.RENDERBUFFER, gl.DEPTH_COMPONENT16, width, height);
 68 	
 69     
 70     //3. Init Frame Buffer
 71     this.framebuffer = gl.createFramebuffer();
 72 	gl.bindFramebuffer(gl.FRAMEBUFFER, this.framebuffer);
 73 	gl.framebufferTexture2D(gl.FRAMEBUFFER, gl.COLOR_ATTACHMENT0, gl.TEXTURE_2D, this.texture, 0);
 74     gl.framebufferRenderbuffer(gl.FRAMEBUFFER, gl.DEPTH_ATTACHMENT, gl.RENDERBUFFER, this.renderbuffer);
 75 	
 76 
 77 	//4. Clean up
 78 	gl.bindTexture(gl.TEXTURE_2D, null);
 79     gl.bindRenderbuffer(gl.RENDERBUFFER, null);
 80     gl.bindFramebuffer(gl.FRAMEBUFFER, null);
 81     
 82     //5. Assign random colors to object in current scene
 83     //for(var i=0, max = this.view.scene._actors.length; i<max; i+=1){
 84     	//this
 85    // }
 86 };

 89 vxlPickerInteractor.prototype.get2DCoords = function(ev){
 90 	var x, y, top = 0, left = 0, obj = this.camera.view.canvas;
 91 
 92 	while (obj && obj.tagName != 'BODY') {
 93 		top += obj.offsetTop;
 94 		left += obj.offsetLeft;
 95 		obj = obj.offsetParent;
 96 	}
 97     
 98     left += window.pageXOffset;
 99     top  += window.pageYOffset;
100  
101 	// return relative mouse position
102 	x = ev.clientX - left;
103 	y = c_height - (ev.clientY - top); //c_height is a global variable that we maintain in codeview.js
104                                        //this variable contains the height of the canvas and it updates dynamically
105                                        //as we resize the browser window.
106 	
107 	return {x:x,y:y};
108 };

110 vxlPickerInteractor.prototype._compare = function(readout, color){
111     vxl.go.console('PickerInteractor: comparing object '+object.alias+' diffuse ('+Math.round(color[0]*255)+','+Math.round(color[1]*255)+','+Math.round(color[2]*255)+') == readout ('+ readout[0]+','+ readout[1]+','+ readout[2]+')');
112     return (Math.abs(Math.round(color[0]*255) - readout[0]) <= 1 &&
113 			Math.abs(Math.round(color[1]*255) - readout[1]) <= 1 && 
114 			Math.abs(Math.round(color[2]*255) - readout[2]) <= 1);
115 };

117 vxlPickerInteractor.prototype.find = function(coords){
118 	
119 	var gl = this.camera.view.renderer.gl;
120 	//read one pixel
121 	var readout = new Uint8Array(1 * 1 * 4);
122 	gl.bindFramebuffer(gl.FRAMEBUFFER, this.framebuffer);
123 	gl.readPixels(coords.x,coords.y,1,1,gl.RGBA,gl.UNSIGNED_BYTE,readout);
124 	gl.bindFramebuffer(gl.FRAMEBUFFER, null);
125 
126     var found = false;
127     var scene = this.view.scene;
128     if (this.hitPropertyCallback == undefined) {alert('The picker needs an object property to perform the comparison'); return;}
129 
130     for(var i = 0, max = scene._actors.length; i < max; i+=1){
131         var ob = scene._actors[i];
132         if (ob.name == 'floor') continue;
133                 
134         var property  = this.hitPropertyCallback(ob);
135     
136         if (this._compare(readout, property)){
137             var idx  = this.plist.indexOf(ob);
138             if (idx != -1){
139                 this.plist.splice(idx,1);
140                 if (this.removeHitCallback){
141                     this.removeHitCallback(ob); 
142                 }
143             }
144             else {
145                 this.plist.push(ob);
146                 if (this.addHitCallback){
147                     this.addHitCallback(ob); 
148                 }
149             }
150             found = true;
151             break;
152         }
153     }
154     draw();
155     return found;
156 };

158 vxlPickerInteractor.prototype.stop = function(){
159     if (this.processHitsCallback != null && this.plist.length > 0){
160         this.processHitsCallback(this.plist);
161     }
162     this.plist = [];
163 };
164 
165 vxlPickerInteractor.prototype.onMouseUp = function(ev){
166 	if(!ev.shiftKey){
167 		this.stop();
168 	}
169 };
170 
171 vxlPickerInteractor.prototype.onMouseDown = function(ev){
172 	var coords = this.get2DCoords(ev);
173 	this.picking = this.find(coords);
174 	if(!this.picking){
175 		this.stop();
176 	}
177 };
178 
179 vxlPickerInteractor.prototype.onMouseMove = function(ev){
180 	this.lastX = this.x;
181 	this.lastY = this.y;
182 	this.x = ev.clientX;
183 	this.y = ev.clientY;
184 	var dx = this.x - this.lastX;
185 	var dy = this.y - this.lastY;
186 	if(this.picking && this.moveCallback){
187 		this.moveCallback(this, dx,dy);
188 	}
189 };
