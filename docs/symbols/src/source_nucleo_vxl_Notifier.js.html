  8 function vxlNotifier(){

 14 vxlNotifier.prototype.addTarget = function(event,t){
 15 	vxl.go.console('vxlNotifier: adding target for event '+event);
 16 	var targetList = this.targetList;
 17 	if (targetList[event]== undefined){
 18 		targetList[event] = [];
 19 	}
 20 	targetList[event].push(t);
 21 };
 22 
 23 
 24 vxlNotifier.prototype.addSource = function(event,src){
 25 	vxl.go.console('vxlNotifier: adding source for event '+event);
 26 	var targetList = this.targetList;
 27 	var sourceList = this.sourceList;
 28 	
 29 	if (sourceList[event]== undefined){
 30 		sourceList[event] = src;
 31 	}
 32 	
 33 	if (targetList[event]== undefined){
 34 		targetList[event] = [];
 35 	}
 36 	
 37 	$(document).bind(event, function(e,event,src,targetList){
 38 		for (var index=0;index<targetList[event].length;index++){
 39 			targetList[event][index].handleEvent(event,src);
 40 		}
 41 	});
 42 };
 43 
 44 vxlNotifier.prototype.fire = function(event){
 45 	var targetList = this.targetList;
 46 	var src = this.sourceList[event];
 47 	vxl.go.console('vxlNotifier: firing ' +event);
 48 	$(document).trigger(event,[event,src,targetList]);
 49 };

 51 vxlNotifier.prototype.getEvents = function(){
 52 	var list = [];
 53 	for (var event in this.sourceList){
 54 		list.push(event);
 55 	}
 56 	return list;
 57 };

 63 		list.push(getObjectName(targets[index]));
 64 	}
 65 	return list;
