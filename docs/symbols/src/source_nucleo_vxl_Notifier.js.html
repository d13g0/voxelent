  4 /**
  5  * <p> 
  6  * Handles asynchronous communication among classes in Voxelent 
  7  * using a publisher-subscriber mechanism
  8  * </p>
  9  * 
 10  * @class
 11  * @constructor
 12  */
 13 function vxlNotifier(){
 14 	this.targetList = {};
 15 	this.sourceList = {};
 16     
 17 };

 19 /**
 20  * <p>Used by any class to declare the events that the class will listen for.</p>
 21   
 22  * @param {Object} list
 23  * @param {Object} receiver
 24  */
 25 vxlNotifier.prototype.subscribe = function(list,receiver){
 26 	if (typeof(list)=='string'){
 27 		this._addTarget(list,receiver);
 28 	}
 29 	else if (list instanceof Array){
 30 		for (var i=0;i<list.length;i+=1){
 31 			this._addTarget(list[i],receiver);
 32 		}
 33 	}
 34 	else {
 35 		throw 'vxlNotifier.receives: this method receives a string or a list of strings'
 36 	}
 37 };

 39 /**
 40  * <p>Used by any class to declare the events that the class will generate</p> 
 41  * @param {Object} list
 42  * @param {Object} sender
 43  */
 44 vxlNotifier.prototype.publish = function(list,sender){
 45 	if (typeof(list)== 'string'){
 46 		this._addSource(list,sender);
 47 	}
 48 	else if (list instanceof Array){
 49 		for (var i=0;i<list.length;i+=1){
 50 			this._addSource(list[i],sender);
 51 		}
 52 	}
 53 	else {
 54 		throw 'vxlNotifier.sends: this method receives a string or a list of strings'
 55 	}
 56 }

 59 /**
 60  * Any class can use this method to tell the notifier that it will listen to 
 61  * a particular event.
 62  * 
 63  * @param {Object} event the event the class will listen to
 64  * @param {Object} target the object that will listen for the event
 65  * @private
 66  */
 67 vxlNotifier.prototype._addTarget = function(event, target){
 68 	vxl.go.console('vxlNotifier: adding target for event '+event);
 69 	var targetList = this.targetList;
 70 	if (targetList[event]== undefined){
 71 		targetList[event] = [];
 72 	}
 73 	targetList[event].push(target);
 74 };

106 /**
107  * <p>Invoked by any class when it needs to emit an event that should be propagated to other
108  * objects in the library</p>
109  * 
110  * <p>The notifier will first verify if the object emitting the event has been authorized to do so.
111  * This is, the object should have registered using <code>publish</code>.
112  * After that, the notifier will retrieve a list of the objects that have registered as listeners of 
113  * the particular event and fires the event to them using JQuery.
114  * </p>
115  *  
116  * @param {Object} event
117  * @param {Object} src
118  */
119 vxlNotifier.prototype.fire = function(event, src){
120     
121     var self = this;
122     
123     function processEvent(){
124     	var targetList = self.targetList;
125         var idx = self.sourceList[event].indexOf(src);
126         if (idx == -1){
127         	throw 'The source '+src+' is not registered to trigger the event '+event+'. Did you use vxlNotifier.publish?';
128         }
129     	vxl.go.console('vxlNotifier: firing ' +event);
130     	
131     	var targets = self.targetList[event];
132     	
133     	for (var index=0;index<targets.length;index++){
134              targets[index].handleEvent(event,src);
135         }
136 	}
137 	processEvent();
138 	setTimeout(function(){processEvent()},0)
139 };

144 /**
145  * Gets a list of the events handled by this vxlNotifier 
146  */
147 vxlNotifier.prototype.getEvents = function(){
148 	var list = [];
149 	for (var event in this.sourceList){
150 		list.push(event);
151 	}
152 	return list;
153 };
