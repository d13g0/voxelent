 17 /**
 18  * <p>Presents a simple interface to communicate with a ESSL (GLSL) program
 19  * This class is responsible for creating, compiling and linking any ESSL program.
 20  * It also has methods to query and set uniforms and attributes belonging to the program
 21  * that is being currently executed in the GPU</P>
 22  * 
 23  * <p>a vxlProgramManager maintains a database of the programs that have been linked to the GPU. 
 24  * This way, program switching is easier as it is not necessary to go through the 
 25  * compilation and linking process every time
 26  * 
 27  * <p>
 28  * The program manager simplifies working with ESSL programs. It provides
 29  * get/set operations for attributes and uniforms and handles internally the location variables
 30  * of these elements which are required to operate with them in the GPU. In other words
 31  * it hides gl.getAttribLocation and gl.getUniformLocation calls.
 32  * </p>
 33  * <p>
 34  * The program manager catches uniforms and only updates the GPU if needed.  
 35  * </p>
 36  * <p>The program manager is available through the <code>pm</code> attribute  of vxlEngine
 37  * </p>
 38  * @class
 39  * @constructor
 40  */
 41 function vxlProgramManager (gl) {
 42     
 43     this._gl                     = gl;
 44     this._registered_programs    = {};
 45     this._programs        = {};
 46     
 47     
 48     this._uniform_map            = {};
 49     this._uniform_types          = {};
 50     
 51     
 52     this._current_program_object   =  null;
 53     this._current_program_ID       = undefined;
 54     this._curr_uniform_loc_map     = {};
 55     this._curr_uniform_cache       = {};
 56     this._curr_attribute_loc_map       = {};
 57     this._enabled_attribute_list   = [];
 58     this._defaults                 = [];
 59  
 60     this._one_time_warning         = false;
 61     this._program_enforced         = false;
 62     
 63 };
 64 
 65 /**
 66  * Verifies whether a program is loaded in the database or not
 67  * @param {String} ID program id
 68  * @returns true if the program is registered, false otherwise
 69  */
 70 vxlProgramManager.prototype._isProgramRegistered = function(ID){
 71     return (this._registered_programs[ID] != undefined);
 72 };

 96 /**
 97  * Loads a program
 98  * @param {String} ID the id of the program to load
 99  */
100 vxlProgramManager.prototype._createProgramObject = function(ID){
101     
102     if (this._isProgramCreated(ID)){
103         return;
104     }
105     
106     var code = this._registered_programs[ID];
107     
108     if (code == undefined){
109         var message = 'vxlProgramManager.loadProgram ERROR: '+
110         ' The program '+ID+' must be registered first!';
111         console.error(message);
112         return;
113     }
114     
115     var gl   = this._gl;
116     var prg  = gl.createProgram();
117     var essl = vxl.def.essl;
118     
119     if (code.VERTEX_SHADER){
120         var vs = this._createShader(essl.VERTEX_SHADER,code.VERTEX_SHADER);
121         gl.attachShader(prg, vs);
122     }
123     
124     if (code.FRAGMENT_SHADER){
125         var fs = this._createShader(essl.FRAGMENT_SHADER,code.FRAGMENT_SHADER);
126         gl.attachShader(prg, fs);
127     }
128     
129     //fix for version 0.89.2 Making sure that the vertex array is ALWAYS the attribute with location 0
130     gl.bindAttribLocation(prg, 0 , essl.VERTEX_ATTRIBUTE);
131     
132     gl.linkProgram(prg);
133      
134     if (!gl.getProgramParameter(prg, gl.LINK_STATUS)) {
135         
136         alert(ID+":\n\n "+gl.getProgramInfoLog(prg));
137         throw("Error linking program "+ID+":\n\n "+gl.getProgramInfoLog(prg));
138     }
139     
140     this._programs[ID] = prg;
141   
142 };

145 /**
146  * Uses a program from the database.
147  * If you are not sure if the program you want to use is in the database then call vxlRenderer.setProgram instead
148  * @param {String} ID the program id
149  * @see vxlRenderer#setProgram
150  */
151 vxlProgramManager.prototype.useProgram = function(ID){
152 
153     var gl = this._gl;
154     var prg = this._programs[ID];
155     
156     if (prg != undefined && prg != null){
157         
158         if (prg == this._current_program_object) return; //don't change if current
159         
160         //gl.linkProgram(esslProgram);
161         gl.useProgram (prg);
162         
163         this._current_program_object = prg;
164         this._current_program_ID = ID;
165         this._parseUniforms();
166         this._one_time_warning  = false;
167     }
168     else{
169         alert("Program: the program " + ID + " has NOT been loaded");
170     }
171 };

173 vxlProgramManager.prototype.releaseProgram = function(){
174      this._enforce = false;

177 /**
178  * Tries to add a new program definition to this renderer
179  * @param {vxlProgram} p_program an instance of a vxlProgram object or one of its descendants
180  */
181 vxlProgramManager.prototype.setProgram = function(p_program, p_force_it){
182     
183     var instance = undefined;
184     //Create a new instance
185     if (typeof p_program == 'function'){
186         instance  = new p_program();
187     }
188     //Use this instance
189     else if (typeof p_program == 'object'){
190         instance = p_program;
191     }
192     else{
193         console.error('vxlProgramManager.setProgram ERROR: '+p_program+' is not an engine');
194         return;
195     }
196     
197     
198     if (this._enforce && instance.ID != this._current_program_id){
199         var message = 'vxlProgramManager.setProgram WARN: '+
200         'Unable to set the program '+instance.ID+'.\n'+
201         'The current program ['+instance.ID+ '] is being enforced\n'+
202         'Please use releaseProgram first.';
203         console.warn(message);
204         return;
205     }
206     
207     this._program_enforced = (p_force_it != undefined && p_force_it == true);
208     //register
209     if (!this._isProgramRegistered(instance.ID)){ this._registerProgram(instance);        }
210     
211     //create
212     if (!this._isProgramCreated(instance.ID)) { this._createProgramObject(instance.ID); }
213     
214     //use
215     this.useProgram(instance.ID);
216     this.clearCache();  //@TODO: what happens when we switch programs back and forth?
217     this.loadDefaults();
218     
219 };
220 
221 vxlProgramManager.prototype.clearCache = function(){
222     this._curr_uniform_cache = {};
223     this._curr_uniform_loc_map ={};
224     this._curr_attribute_loc_map = {};
225 };

229 /**
230  * Loads the uniform defaults for the current program
231  */
232 vxlProgramManager.prototype.loadDefaults = function(){
233     var code = this._registered_programs[this._current_program_ID];
234    
235     if ('DEFAULTS' in code){
236     
237         var defaults = code.DEFAULTS;
238         
239         
240         for(var u in defaults){
241             this.setUniform(u,defaults[u]);
242         }
243     }
244     //overriding defaults
245     var defaults = this._defaults[this._current_program_ID];
246     if (defaults != undefined){
247   
248         for (var u in defaults){
249             this.setUniform(u,defaults[u]);
250         }
251     }
252    
253 };
254 
255 /**
256  * Overrides defaults by hand 
257  */
258 vxlProgramManager.prototype.setDefault = function(programID, uniformName, value){
259     
260     if (this._defaults[programID] == undefined){
261         this._defaults[programID] = {};
262     }
263   
264     this._defaults[programID][uniformName] = value;
265     
266     //Overriding behaviour
267     if (programID == this._current_program_ID){
268         this.setUniform(uniformName, value);
269     }
270 };
271 
272 
273 /**
274  * Overrides defaults by hand 
275  */
276 vxlProgramManager.prototype.getDefault = function(programID, uniformName){
277     
278     if (this._defaults[programID] == undefined){
279         return undefined;
280     }
281   
282     return this._defaults[programID][uniformName];
283 };

296 /**
297  * Sets a uniform. Caches the uniform location.
298  * 
299  * Uses polymorphism to make the programmers life happier
300  * @param {String} p_uniform_id name
301  * @param {Object} p_value the uniform value 
302  */
303 vxlProgramManager.prototype.setUniform = function(p_uniform_id, p_value, hint){
304     var gl                  = this._gl;
305     var prg          		= this._current_program_object;
306     var uniform_list  		= this._uniform_map[this._current_program_ID];
307     var uniform_loc_map	    = this._curr_uniform_loc_map;
308     var uniform_cache 		= this._curr_uniform_cache;
309     var uniform_types       = this._uniform_types[this._current_program_ID];
310     var loc                 = undefined;
311     var reset               = false;
312     
313     if (uniform_list.indexOf(p_uniform_id) == -1){
314         console.warn('vxlProgramManager.setUniform: the uniform '+p_uniform_id+' is not defined for the program '+this._current_program_ID);
315         return;
316     }
317     
318     loc = uniform_loc_map[p_uniform_id];
319     
320     if (loc == undefined){  
321         loc = gl.getUniformLocation(prg,p_uniform_id);  
322         uniform_loc_map[p_uniform_id] = loc;     
323     }
324     
325     var cached_value = uniform_cache[p_uniform_id];
326     var type = uniform_types[p_uniform_id];
327     
328     if (cached_value == undefined){
329         reset = true;
330     }
331     else{ 
332        switch(type){ 
333             case 'sampler2D':
334             case 'float':
335             case 'int':
336             case 'bool': reset = (cached_value !== p_value); break;
337             case 'mat4':
338                     reset  = (
339                         (p_value[0] !== cached_value[0]) ||
340                         (p_value[1] !== cached_value[1]) ||
341                         (p_value[2] !== cached_value[2]) ||
342                         (p_value[3] !== cached_value[3]) ||
343                         (p_value[4] !== cached_value[4]) ||
344                         (p_value[5] !== cached_value[5]) ||
345                         (p_value[6] !== cached_value[6]) ||
346                         (p_value[7] !== cached_value[7]) ||
347                         (p_value[8] !== cached_value[8]) ||
348                         (p_value[9] !== cached_value[9]) ||
349                         (p_value[10] !== cached_value[10]) ||
350                         (p_value[11] !== cached_value[11]) ||
351                         (p_value[12] !== cached_value[12]) ||
352                         (p_value[13] !== cached_value[13]) ||
353                         (p_value[14] !== cached_value[14]) ||
354                         (p_value[15] !== cached_value[15])); break;
355             case 'vec3':
356                     reset = (
357                         (p_value[0] !== cached_value[0]) ||
358                         (p_value[1] !== cached_value[1]) ||
359                         (p_value[2] !== cached_value[2])); break;
360             case 'vec4':
361                     reset = (
362                         (p_value[0] !== cached_value[0]) ||
363                     (p_value[1] !== cached_value[1]) ||
364                     (p_value[2] !== cached_value[2]) ||
365                     (p_value[3] !== cached_value[3])); break;
366             default:
367                 reset = true;
368         }
369     }
370    
371     if (reset){
372         switch(type){ 
373             case 'float':
374             case 'int':
375             case 'bool': uniform_cache[p_uniform_id] = p_value; break;
376             case 'mat4': uniform_cache[p_uniform_id] = mat4.clone(p_value); break;
377             case 'mat3': uniform_cache[p_uniform_id] = mat3.clone(p_value); break;
378             case 'vec4': uniform_cache[p_uniform_id] = vec4.clone(p_value); break;
379             case 'vec3': uniform_cache[p_uniform_id] = vec3.clone(p_value); break;
380             case 'vec2': uniform_cache[p_uniform_id] = vec2.clone(p_value); break;
381             case 'sampler2D': uniform_cache[p_uniform_id] = p_value; break;
382             default: alert('error: type unknown cannot update uniform cache');
383         }
384         this._setPolymorphicUniform(p_uniform_id, loc, p_value, hint);
385     }
386 
387 };

413 /**
414  * Enables a vertex attribute array
415  * @param {String} name the name of the attribute array to enable
416  */
417 vxlProgramManager.prototype.enableAttribute = function(name){
418     
419    if (this._enabled_attribute_list.indexOf(name) != -1) return; //Speeds up
420      
421    var a = this._getAttributeLocation(name);
422    this._gl.enableVertexAttribArray(a);
423    this._enabled_attribute_list.push(name);
424 };
425 
426 /**
427  * Disables a vertex attribute array
428  * @param {String} name the name of the attribute array to disable
429  * 
430  */
431 vxlProgramManager.prototype.disableAttribute = function(name){
432     if (name == undefined) return; //@TODO: WARNING?
433     var idx = this._enabled_attribute_list.indexOf(name);
434     if ( idx>=0){
435         var loc = this._getAttributeLocation(name);
436         this._gl.disableVertexAttribArray(loc);
437         this._enabled_attribute_list.splice(idx,1);
438     }
439 };

441 /**
442  * Creates a WebGL shader
443  * 
444  * @private This method is private.
445  */
446 vxlProgramManager.prototype._createShader = function(type,code){
447     var gl      = this._gl;
448     var shader = null;
449     
450     if (type == vxl.def.essl.VERTEX_SHADER){
451         shader = gl.createShader(gl.VERTEX_SHADER);
452     }
453     else if (type == vxl.def.essl.FRAGMENT_SHADER){
454         shader = gl.createShader(gl.FRAGMENT_SHADER);
455     }
456     
457     if (code == undefined || code == null){
458         alert('Error getting the code for shader of type ' + type);
459     }
460     
461     gl.shaderSource(shader, code);
462     gl.compileShader(shader);
463 
464     if (!gl.getShaderParameter(shader, gl.COMPILE_STATUS)) {
465         alert(type+":\n\n "+gl.getShaderInfoLog(shader));
466         throw("Error compiling shader "+type+":\n\n "+gl.getShaderInfoLog(shader));
467     }
468     
469     return shader;
470 };

473 /**
474  * Parses uniforms
475  * This method is private
476  * @private
477  * 
478  */
479 vxlProgramManager.prototype._parseUniforms = function(id){
480     
481     vs = this._registered_programs[this._current_program_ID].VERTEX_SHADER;
482     fs = this._registered_programs[this._current_program_ID].FRAGMENT_SHADER;
483     /*@TODO: look for a way to retrieve uNames directly from the parsing of the shaders
484     this should simplify the structure of the JSON file representing the program*/
485     uNames = this._registered_programs[this._current_program_ID].UNIFORMS;
486     
487     uTypes = {};
488     
489     
490     for (var i=0;i< uNames.length; i++){
491         var uniformID = uNames[i];
492         var rex = new RegExp('uniform\\s+\\w+\\s'+uniformID,'g');
493         
494         if (vs.search(rex) != -1){
495             uTypes[uniformID] = vs.substring(vs.search(rex),vs.length).substring(0,vs.indexOf(';')).split(' ')[1];
496         }
497         
498         else if(fs.search(rex) != 1){
499             uTypes[uniformID] = fs.substring(fs.search(rex),fs.length).substring(0,fs.indexOf(';')).split(' ')[1];
500         }
501         
502         else{
503             alert('Program: In the program '+this._current_program_ID+' the uniform '+uniformID+' is listed but not used');
504         }
505     }
506     
507     
508     this._uniform_map[this._current_program_ID] = uNames;
509     this._uniform_types[this._current_program_ID] = uTypes; 
510 };

512 /**
513  * Obtains an attribute location
514  * This method is private
515  * @param {String} name
516  * @private
517  */
518 vxlProgramManager.prototype._getAttributeLocation = function(name){
519     
520     var loc = this._curr_attribute_loc_map[name];
521     if (loc != undefined) return loc;
522      
523     loc = this._gl.getAttribLocation(this._current_program_object,name);
524     if (loc == -1){
525         console.error('vxlProgramManager._getAttributeLocation ERROR: the attribute '+name+''+
526         'could not be located');
527         loc = undefined;
528     }
529     else{
530         this._curr_attribute_loc_map[name] = loc;
531      }
532     return loc;
533 };

535 /**
536  * This is one of the jewels of Voxelent. Based on the information contained in the 
537  * program database, it will do the appropriate gl call to set the uniform
538  * This method is private. Use setUniform instead.
539  * @see vxlProgramManager#setUniform
540  * @private 
541  */
542 vxlProgramManager.prototype._setPolymorphicUniform = function(uniformID, locationID,value,hint){
543 
544 	//In the extend of what it is reasonable,
545 	//We cross check GLSL type information with actual javascript variable types 
546 	//to make the right calls
547 	//hint allows better casting of int and float values. If not specified default is float
548     
549     var gl = this._gl;
550     var glslType = this._uniform_types[this._current_program_ID][uniformID];
551     
552     if (glslType == 'bool'){
553     	//if (typeof value != 'boolean') { 
554     	//	vxl.go.console('Program: the uniform '+uniformID+' is defined as bool in GLSL. However the JS variable is not');
555     	//}/
556         gl.uniform1i(locationID,value);
557         return;
558     }
559     
560     else if (glslType == 'float'){
561     	gl.uniform1f(locationID,value);
562     	return;
563     }
564     
565     else if (glslType == 'int' || glslType == 'sampler2D'){
566         gl.uniform1i(locationID,value);
567         return;
568     }
569     
570     else if (glslType == 'mat4'){    
571         gl.uniformMatrix4fv(locationID,false,value);
572         return;
573     }
574     
575     
576     else if (value instanceof Array || value instanceof Float32Array){ //vec2, vec3, vec4
577         
578         /*If we receive a uniform of length 3 but the type is length 4
579          * complete with 1.0
580          *  This is a hack that needs to be revisited....
581          * 
582          */ 
583          
584         if (value.length == 3 && glslType == 'vec4'){
585              value[3] = 1.0;
586              if (!this._one_time_warning){
587                  alert('The uniform '+uniformID+' has only 3 components but voxelent needs 4. This is a one time warning');
588                  this._one_time_warning = true;
589              }
590         }
591         
592         if (hint  == 'int'){
593             switch(value.length){
594                 case 1: { gl.uniform1iv(locationID,value); break; };
595                 case 2: { gl.uniform2iv(locationID,value); break; };
596                 case 3: { gl.uniform3iv(locationID,value); break; };
597                 case 4: { gl.uniform4iv(locationID,value); break; };
598                 default: alert('ERROR');
599             }
600        }
601        else{
602             switch(value.length){
603                 case 1 : { gl.uniform1fv(locationID,value); break; }
604                 case 2 : { gl.uniform2fv(locationID,value); break; }
605                 case 3 : { gl.uniform3fv(locationID,value); break; }
606                 case 4 : { gl.uniform4fv(locationID,value); break; }
607                 default: alert('ERROR');
608             }
609        }
610     }
611     
612     else {
613     	alert('Program: ERROR. The uniform  '+uniformID+ ' could not be mapped');
614     }
615 };
