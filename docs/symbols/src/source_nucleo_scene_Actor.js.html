 18 /**
 19  * <p>
 20  * An actor is a representation of a model in Voxelent. Actors can cache model properties 
 21  * and modified them. This is useful when there are several actors based in the same model
 22  * but each one of them needs to have a different version of any given model property (i.e. color)
 23  * </p>
 24  * <p> 
 25  * To propagate one change for all the actors based in the same model, the setProperty method
 26  * should be invoked by setting the third parameter (scope) like this:
 27  * </p>
 28  * 
 29  * <pre class="prettyprint">
 30  * var actor = vxl.c.scene.getActorByName('example');
 31  * actor.setProperty('color',[1.0,0.0,0.0], vxl.def.model)
 32  * </pre>
 33  * 
 34  * <p>If the change should be local (for just that actor)  then you should write:</p>
 35  * 
 36  * <pre class="prettyprint">
 37  * var actor = vxl.c.scene.getActorByName('example');
 38  * actor.setProperty('color',[1.0,0.0,0.0], vxl.def.actor)
 39  * </pre>
 40  * 
 41  * <p> Or simply </p>
 42  *  
 43  * <pre class="prettyprint">
 44  * var actor = vxl.c.scene.getActorByName('example');
 45  * actor.setProperty('color',[1.0,0.0,0.0])
 46  * </pre>
 47  * @class Actors represent models (assets) in a Scene
 48  * @constructor
 49  */
 50 function vxlActor(model){
 51   
 52   this._bb            = [0, 0, 0, 0, 0, 0];
 53   this._position 	  = vec3.fromValues(0, 0, 0);
 54   this._translation   = vec3.fromValues(0, 0, 0);
 55   this._centre        = vec3.fromValues(0, 0, 0);
 56   this._scale 		  = vec3.fromValues(1, 1, 1);
 57   this._rotation 	  = vec3.fromValues(0, 0, 0);
 58   this._matrix        = mat4.create();
 59   this._matrix_world  = mat4.create();
 60   this._matrix_normal = mat4.create();
 61   this._dirty         = false;
 62   this._picking       = vxl.def.actor.picking.DISABLED;
 63   this._pickingCallback = undefined;
 64   this._pickingColor    = undefined; //used when the picking is vxl.def.actor.picking.OBJECT
 65   this._trackingCameras = [];
 66   
 67   this.UID = vxl.util.generateUID();
 68   this.scene        = undefined;
 69   this.clones       = 0;
 70   
 71   this.mode         = vxl.def.actor.mode.SOLID;
 72   this.cull         = vxl.def.actor.cull.NONE;
 73   this.visible      = true;
 74  
 75   this.mesh         = undefined;
 76   
 77   this.material     = new vxlMaterial();
 78   this.renderable   = undefined;
 79   
 80   this._bb_disabled = false; //to accelerate animations if we dont care about bb.
 81                              //@TODO: find another way to optimize this.
 82   
 83   if (model){
 84   	this.model 	    = model;
 85   	this.name 	    = model.name;
 86   	this.mode       = model.mode;
 87   	this._bb        = model.bb.slice(0);
 88   	this._centre    = vec3.clone(model.centre);
 89   	this.material.getFrom(model);
 90   	
 91   	if (model.type == vxl.def.model.type.BIG_DATA){
 92   	    this.renderable = new vxlRenderable(this);
 93   	}
 94   }
 95   else{
 96       this.model = new vxlModel();
 97   }
 98   
 99   var e = vxl.events;
100   vxl.go.notifier.publish(
101       [
102         e.ACTOR_MOVED,
103         e.ACTOR_SCALED,
104         e.ACTOR_ROTATED,
105         e.ACTOR_CHANGED_COLOR,
106         e.ACTOR_CHANGED_SHADING,
107       ], this);
108 };

120 /**
121  * Sets the position of this actor. 
122  * 
123  * 
124  * @param {Number, Array, vec3} x it can be the x coordinate, a 3-dimensional Array or a vec3 (glMatrix)
125  * @param {Number} y if x is a number, then this parameter corresponds to the y-coordinate
126  * @param {Number} z if x is a number, then this parameter corresponds to the z-coordinate
127  */
128 vxlActor.prototype.setPosition = function (x,y,z){
129     var np = vxl.util.createVec3(x,y,z);
130     vec3.subtract(this._translation, np, this._position);
131     this._position = np;  
132     
133     var m = this._matrix;
134     m[12] = this._position[0];
135     m[13] = this._position[1];
136     m[14] = this._position[2];
137     m[15] = 1;
138     if (!this._bb_disabled) {
139         this._computeBoundingBox();
140     }
141     
142     this._notifyTrackingCameras();
143     vxl.go.notifier.fire(vxl.events.ACTOR_MOVED, this);
144     
145     return this;
146 };

149 /**
150  * Translates the actor by a given vector 
151  * 
152  * @param {Number, Array, vec3} x it can be the x coordinate, a 3-dimensional Array or a vec3 (glMatrix)
153  * @param {Number} y if x is a number, then this parameter corresponds to the y-coordinate
154  * @param {Number} z if x is a number, then this parameter corresponds to the z-coordinate
155  */
156 vxlActor.prototype.translate = function (x,y,z){
157     this._translation = vxl.util.createVec3(x,y,z); 
158     
159     var m = this._matrix;
160     mat4.translate(m, m,this._translation);
161     this._position = vec3.fromValues(m[12], m[13], m[14]);
162     
163     if (!this._bb_disabled) { 
164         this._computeBoundingBox(); 
165     }
166     this._notifyTrackingCameras();
167     vxl.go.notifier.fire(vxl.events.ACTOR_MOVED, this);
168     
169     return this;
170 };

225 /**
226  * Scales this actor. 
227  * @param {Number, Array, vec3} s the scaling factor. The scaling factor is applied in all axes.
228  *
229  */
230 vxlActor.prototype.setScale = function(s,a,b){
231     if (s == 0 && a == undefined && b==undefined) return;
232     
233     
234     if (typeof(s)=="number" && a == undefined && b == undefined){
235         this._scale = vxl.util.createVec3(s,s,s);
236     }
237     else{
238         this._scale = vxl.util.createVec3(s,a,b);
239     }
240     
241     var m = this._matrix;
242     mat4.scale(m, m, this._scale);
243     if (!this._bb_disabled){
244         this._computeBoundingBox();
245     }
246     vxl.go.notifier.fire(vxl.events.ACTOR_SCALED, this);
247     
248     return this;
249 };

251 /**
252  * Adds a tracking camera
253  * 
254  * @param{vxlCamera} camera the tracking camera
255  */
256 vxlActor.prototype.addTrackingCamera = function(camera){
257     if (camera instanceof vxlCamera && camera.type !=vxl.def.camera.type.TRACKING){
258         throw "vxlActor._addTrackingCamera ERROR: the selected camera is not set to tracking"
259     }
260     else if (camera instanceof vxlCamera){
261         this._trackingCameras.push(camera);
262     }
263     else{
264         throw "vxlActor._addTrackingCamera ERROR: the object passed as a parameter is not a vxlCamera"
265     }

287 /**
288  * Computes the current bounding box for this actor.
289  * This method is called on demand by the scene or any other object. 
290  * The bounding box is NOT automatically recalculated when 
291  * moving or scaling an actor for performance reasons.
292  *  
293  */
294 vxlActor.prototype._computeBoundingBox = function(){
295 
296     var vs  = this.model.vertices;
297     var vsT = [];
298     var T = this._matrix;
299     
300     for(var i=0;i<vs.length;i=i+3){
301         var x = vxl.util.createVec3(vs[i],vs[i+1],vs[i+2]);
302         vec3.transformMat4(x, x, T);
303         vsT.push(x[0], x[1], x[2]);    
304     }
305     
306     var bbA = [vsT[0],vsT[1],vsT[2],vsT[0],vsT[1],vsT[2]];
307     
308     for(var i=0;i<vsT.length;i=i+3){
309         bbA[0] = Math.min(bbA[0],vsT[i]);
310         bbA[1] = Math.min(bbA[1],vsT[i+1]);
311         bbA[2] = Math.min(bbA[2],vsT[i+2]);
312         bbA[3] = Math.max(bbA[3],vsT[i]);
313         bbA[4] = Math.max(bbA[4],vsT[i+1]);
314         bbA[5] = Math.max(bbA[5],vsT[i+2]);
315     }
316     
317      
318     this._bb = bbA;
319 };

321 /**
322  * Returns an array with the bounding box vertices. Good for rendering the transformed
323  * bounding box (after geometric transformations) 
324  * @returns {Array} a 8-element array with the vertices that constitute the actor bounding box
325  */
326 vxlActor.prototype.getBoundingBoxVertices = function(){
327     var b = this._bb; 
328     return [
329         b[0], b[1], b[2],
330         b[0], b[4], b[2],
331         b[3], b[4], b[2],
332         b[3], b[1], b[2],
333         b[0], b[1], b[5],
334         b[0], b[4], b[5],
335         b[3], b[4], b[5],
336         b[3], b[1], b[5] 
337         ];
338 };

415 /**
416  * If the property exists, then it updates it
417  * @param {String} property name of the property 
418  * @param {Object} value  value to be set
419  * @param {String} scope indicates if the change is made at the actor level or at the model level
420  * valid values for scope are vxl.def.model and vxl.def.actor
421  * @TODO: if the property is position or scale then call the respective methods from here
422  */
423 vxlActor.prototype.setProperty = function(property, value, scope){
424     
425     
426     if (scope == vxl.def.actor || scope == undefined || scope == null){
427 
428         switch (property){
429             case 'position': this.setPosition(value);  break;
430             case 'scale':    this.setScale(value);     break;
431             case 'color':    this.setColor(value);     break;
432             case 'shading':  this.setShading(value);   break;
433             case 'texture':  this.setTexture(value);   break;
434             case 'opacity':  this.setOpacity(value);   break;
435             case 'shininess':this.setShininess(value); break;
436             default: this[property] = value; break;
437         }
438   
439 		vxl.go.console('Actor: The actor '+this.name+' has been updated. ['+property+' = '+value+']');
440 	}
441 	else if(scope == vxl.def.model){
442 		this.model[property] = value;
443 		vxl.go.console('vxlActor: The model '+this.model.namname+' has been updated. ['+property+' = '+value+']');
444 	}
445 	else{
446 		throw('vxlActor.setProperty. Scope:' + scope +' is not valid');
447 	}
448 	
449 	return this;
450 	
451 };

465 /**
466  * Returns an actor property if that property exists in the actor. Otherwise it will search 
467  * in the model. This method is used by the renderer. There are some cases where actors have local changes
468  * that are not reflected in the model. In these cases the renderer should pick the actor property
469  * over the model property
470  * @param {String} property the property name
471  * @returns {Object} the property or undefined if the property is not found 
472  */
473 vxlActor.prototype.getProperty = function(property){
474     
475     if (property == 'color'){
476         return this.materia.diffuse; //there's no real 'color' property.
477     }
478 	else if (this.hasOwnProperty(property)) {
479 		return this[property];
480 	}
481 	else if (this.material.hasOwnProperty(property)){
482 	   return this.material[property];   
483 	}
484 	else if (this.model.hasOwnProperty(property)){
485 		return this.model[property];
486 	}
487 	else {
488 		return undefined;
489 	}
490 };

492 /**
493  * Estimates the current position as
494  * the center of the current bounding box. 
495  * This method does not update the internal position of the actor
496  * it only returns an estimate based on the location of its bounding box.
497  * Is there really any use for this? I forgot..!
498  * 
499  * @private
500  */
501 vxlActor.prototype.computePosition = function(){
502 	bb = this._bb;
503 	var cc = vec3.create();
504 	
505 	cc[0] = (bb[3] + bb[0]) /2;
506 	cc[1] = (bb[4] + bb[1]) /2;
507 	cc[2] = (bb[5] + bb[2]) /2;
508 		
509 	cc[0] = Math.round(cc[0]*1000)/1000;
510 	cc[1] = Math.round(cc[1]*1000)/1000;
511 	cc[2] = Math.round(cc[2]*1000)/1000;
512 	
513 	return vec3.clone(cc); 
514 };

540 /**
541 * Sets the lookup table for this actor.
542 * This method will only succeed if the model that this actor represents has scalars 
543 * @param {String} lutID the lookup table id. See {@link vxl.def.lut} for currently supported ids.
544 * @param {Number} min lowest value for interpolation
545 * @param {Number} max highest value for interpolation
546 */
547 vxlActor.prototype.setLookupTable = function(lutID,min,max){
548 	
549 	if (this.model.scalars == undefined) return;
550 	
551 	var self = this;
552 	
553 	function scheduledSetLookupTable(scalars){
554         var lut = vxl.go.lookupTableManager.get(lutID);
555         self.material.colors  = lut.getColors(self.model.scalars,min,max);
556         
557         //if(this.mesh){
558         //    this.mesh. //update mesh with vertex colors it may require access to the original index array
559        // }
560        
561        if (self.model.type == vxl.def.model.type.BIG_DATA){
562            self.renderable.update();
563        }
564 	}
565 	
566 	//Given that obtaining the colors can be a time consume op, it is deferred here.
567 	setTimeout(function(){scheduledSetLookupTable()},0);
568 	
569 	return this; 
570 };

600 /**
601  * Duplicates this actor.
602  * 
603  * <p>The model property of the actor (vxlModel) is shared with the new instance</p>
604  * 
605  * <p>The actor level properties are copied by VALUE.</p>
606  * 
607  * If a cloned actor modifies his internal model, any other actor that shares the model will be
608  * affected. A cloned actor however can have different position, colors, etc. (actor level properties)
609  * 
610  * The returned actor is not added to the scene automatically. It is up the
611  * programmer to determine the scene the cloned actor needs to be added to if at all.
612  * 
613  * The clone does not keep any information about the current position or rotation of the original
614  * actor.
615  * 
616  * @see vxlModel
617  * @returns {vxlActor} an actor 
618  */
619 vxlActor.prototype.clone = function(){
620     this.clones++;
621 	
622 	//create clone
623 	var clone = new vxlActor(this.model);
624 	clone.name     += '-'+this.clones; 
625 	
626 	//copy the material
627 	clone.material = this.material.clone();
628 	
629 	return clone;
630 };

632 /**
633  * 
634  * @param {String} type one of the possible values for vxl.def.actor.picking
635  * @param {Object} callback a function that is invoked when a picking event occurs. This parameter is 
636  * required if the type (first argument) is different from vxl.def.actor.picking.DISABLED 
637  * the callback receives an actor object to operate over it.
638  */
639 vxlActor.prototype.setPicker = function(type, callback){
640     this._picking = type;
641     
642     switch(type){
643         case vxl.def.actor.picking.DISABLED: 
644             this._pickingCallback = undefined;
645             this.mesh = undefined;
646             this.renderable = undefined;
647             this.setVisualizationMode(vxl.def.actor.mode.SOLID); 
648             break;
649         
650         case vxl.def.actor.picking.CELL: 
651             
652             if (this.mesh == undefined){
653                 this.mesh = new vxlMesh(this); 
654                 this.mesh.setColor(this.material.diffuse);
655                 this.renderable = new vxlRenderable(this);
656                 this.setVisualizationMode(vxl.def.actor.mode.FLAT);           
657             };
658             
659             this._pickingCallback = callback;
660              
661             break;
662         case vxl.def.actor.picking.OBJECT:
663             this._pickingColor = vxl.go.picker.getColorFor(this);
664             this._pickingCallback = callback;
665             break;
666     }
667     
668     if (this._picking  != vxl.def.actor.picking.DISABLED){
669         var i = this.scene.views.length;
670         while(i--){
671             //@TODO: do we want this for all the views?
672             var r = this.scene.views[i].renderer;
673             if(!r.isOffscreenEnabled()){
674                 r.enableOffscreen();
675             }
676         }
677     }
678     //@TODO: Disable when there are no pickable actors in the scene.
679     
680     return this;
681 };
