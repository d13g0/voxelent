 17 /**
 18  * @class Manages the axis, the bounding box and the floor
 19  * These are auxiliary objects or 'toys'
 20  */
 21 function vxlSceneToys(scn){
 22     
 23     this.scene = scn;
 24     this.list   = [];
 25 
 26 	this.axis 				= new vxlAxis();
 27 	this.boundingbox 		= new vxlBoundingBox();
 28 	this.floor  			= new vxlFloor();
 29 	
 30 	this.list.push(this.axis);
 31     this.list.push(this.boundingbox);
 32     this.list.push(this.floor);
 33 };
 34 
 35 /**
 36  * Updates the toys according to information from the scene
 37  * 
 38  */
 39 vxlSceneToys.prototype.update = function(){
 40     this.axis.setCentre(this.scene.centre);
 41     this.boundingbox.setBoundingBox(this.scene.bb);
 42 };
 43 /**
 44  * Renders the toys
 45  * @param {vxlRenderer} r the renderer
 46  */
 47 vxlSceneToys.prototype.render = function(r){
 48     for(var t = 0, max = this.list.length; t < max; t +=1){
 49         this.list[t]._allocate(r);
 50         this.list[t]._render(r);
 51     }
 52 };
