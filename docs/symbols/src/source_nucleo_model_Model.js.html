 17 /**
 18  * Models are totally independent of views and of the rendering process
 19  * @class Represents a geometric object. A model is represented by an actor in a scene.
 20  * @constructor
 21  * @param {String} name the name for this model
 22  * @param {Object} JSON_OBJECT the JSON Object that defines this model (Optional)
 23  * @see vxlModel#load
 24  * @author Diego Cantor
 25  */
 26 function vxlModel(name, JSON_OBJECT){
 27     this.UID        = vxl.util.generateUID();
 28 	this.name       = name;
 29 	this.indices 	= [];
 30 	this.vertices 	= [];
 31 	this.scalars 	= undefined;
 32 	this.diffuse	= undefined;
 33 	this.ambient    = undefined;
 34 	this.specular   = undefined;
 35 	this.shininness = undefined;
 36 	this.normals 	= undefined;
 37 	this.wireframe 	= undefined;
 38 	this.bb         = [0,0,0,0,0,0];
 39     this.centre     = [0,0,0,0,0,0];
 40 	this.mode       = vxl.def.actor.mode.SOLID;
 41 	this.image      = undefined;
 42 	this.uri        = undefined;
 43 	this.colors     = undefined;
 44 	this.type       = vxl.def.model.type.SIMPLE;
 45 	this.renderable = undefined;
 46    
 47     if (JSON_OBJECT != undefined){
 48         this.load(this.name, JSON_OBJECT);
 49     }
 50     
 51     this.setType(this.type);
 52 };
 53 
 54 /**
 55  * Sets the type of model. A renderable object is created and associated
 56  * with this model in case the type is MESH or BIG_DATA
 57  * 
 58  * @param {vxl.def.model.type} type the type of model
 59  */
 60 vxlModel.prototype.setType = function(type){
 61     
 62     if (this.indices.max() < vxl.def.model.MAX_NUM_INDICES && 
 63         type == vxl.def.model.type.BIG_DATA){
 64             throw('The model is not big enough to be of BIG_DATA type. Max index found: '+this.indices.max());
 65         }
 66     
 67     this.type = type;  
 68 };

 78 /**
 79  * Populates this model with the JSON_OBJECT (JSON object)
 80  * @param {String} nm the name given to this model
 81  * @param {Object} JSON_OBJECT the JSON object that describes the model
 82  */
 83 vxlModel.prototype.load = function(nm,JSON_OBJECT){
 84     if (nm == undefined){
 85         throw 'vxlModel.load ERROR: the object must have a name';
 86     }
 87 	this.name		= nm.replace(/\.[^/.]+$/, "");
 88 	if (JSON_OBJECT.name != null){ //if the name is defined in the JSON object, then use it
 89 		this.name = JSON_OBJECT.name;
 90 	}
 91 	
 92 	//Load all properties
 93 	for(i in JSON_OBJECT){
 94 		this[i] = JSON_OBJECT[i];
 95 	}
 96 	
 97 	this.update();
 98 
 99 };

101 /**
102  * Update model state when the model has changed (changes in vertices or indices) 
103  */
104 vxlModel.prototype.update = function(){
105         //Now minimal checks
106     if (this.vertices == undefined){
107         alert('The model '+ this.name+' does not have vertices. Impossible to render!');
108     }
109     
110 
111     if(this.normals == undefined
112         && this.indices != undefined
113         && this.mode != vxl.def.actor.mode.LINES) //@TODO: explain that mode is representation mode. (change name?)
114     {
115         this.computeNormals();
116     }
117     
118      
119     if (this.wireframe == undefined){
120         this.computeWireframeIndices();
121     }
122     
123     if (this.mode == undefined){
124         this.mode = vxl.def.actor.mode.SOLID;
125     }
126     
127     if (this.texture != undefined){
128         this.mode = vxl.def.actor.mode.TEXTURED;
129     }
130     
131     this.computeBoundingBox();
132     
133     
134     if (this.type == vxl.def.model.type.SIMPLE && this.indices.max() > vxl.def.model.MAX_NUM_INDICES){
135         this.setType(vxl.def.model.type.BIG_DATA);
136         console.info('the model '+this.name+' type is BIG_DATA');
137     }
138     
139 };

141 /**
142  * Calculates the normals for this model in case that the JSON object does not include them
143  * 
144  * 
145  */
146 vxlModel.prototype.computeNormals = function(){
147   //face normal calculation
148     var vs = this.vertices;
149 	var ind = this.indices;
150 	var x=0; 
151     var y=1;
152 	var z=2;
153 	
154 	var ns = [];
155 	for(var i=0;i<vs.length;i=i+3){ //for each index, initialize normal x, normal y, normal z
156 		ns[i+x]=0.0;
157 		ns[i+y]=0.0;
158 		ns[i+z]=0.0;
159 	}
160 	
161 	for(var i=0;i<ind.length;i=i+3){ //we work on triads of vertex to calculate normals so i = i+3 (i = indices index)
162 		var v1 = [];
163 		var v2 = [];
164 		var normal = [];	
165 		//p2 - p1
166 		v1[x] = vs[3*ind[i+2]+x] - vs[3*ind[i+1]+x];
167 		v1[y] = vs[3*ind[i+2]+y] - vs[3*ind[i+1]+y];
168 		v1[z] = vs[3*ind[i+2]+z] - vs[3*ind[i+1]+z];
169 		//p0 - p1
170 		v2[x] = vs[3*ind[i]+x] - vs[3*ind[i+1]+x];
171 		v2[y] = vs[3*ind[i]+y] - vs[3*ind[i+1]+y];
172 		v2[z] = vs[3*ind[i]+z] - vs[3*ind[i+1]+z];
173 		//cross product
174 		normal[x] = v1[y]*v2[z] - v1[z]*v2[y];
175 		normal[y] = v1[z]*v2[x] - v1[x]*v2[z];
176 		normal[z] = v1[x]*v2[y] - v1[y]*v2[x];
177 		
178 		for(j=0;j<3;j++){ //update the normals of the triangle
179 			ns[3*ind[i+j]+x] =  ns[3*ind[i+j]+x] + normal[x];
180 			ns[3*ind[i+j]+y] =  ns[3*ind[i+j]+y] + normal[y];
181 			ns[3*ind[i+j]+z] =  ns[3*ind[i+j]+z] + normal[z];
182 		}
183 	}
184 		
185 	//normalize the result
186 	for(var i=0;i<vs.length;i=i+3){ //the increment here is because each vertex occurs with an offset of 3 in the array (due to x, y, z contiguous values)
187 	
188 	    var nn=[];
189 		nn[x] = ns[i+x];
190 		nn[y] = ns[i+y];
191 		nn[z] = ns[i+z];
192 		
193 		var len = Math.sqrt((nn[x]*nn[x])+(nn[y]*nn[y])+(nn[z]*nn[z]));
194 		if (len == 0) len = 1.0;
195 		
196 		nn[x] = nn[x]/len;
197 		nn[y] = nn[y]/len;
198 		nn[z] = nn[z]/len;
199 		
200 		ns[i+x] = nn[x];
201 		ns[i+y] = nn[y];
202 		ns[i+z] = nn[z];
203 	}
204 	this.normals = ns;
205 };

207 /**
208  * Flips the normals 
209  */
210 vxlModel.prototype.flipNormals = function(){
211     var ns = this.normals;
212     
213     if (ns == undefined) return; //no normals
214     
215     for (var i=0; i<ns.length; i+=1){
216         ns[i] = -ns[i];
217     }
218 };

220 /**
221  * Generate the wireframe indices using the model indices
222  */  
223 vxlModel.prototype.computeWireframeIndices = function(){
224 	var ind = this.indices;
225     var wfi = [];
226 	var j = 0;
227 	for(var i=0; i<ind.length; i=i+3){
228 	   wfi[j] = ind[i];
229 	   wfi[j+1] = ind[i+1];
230 	   wfi[j+2] = ind[i+1];
231 	   wfi[j+3] = ind[i+2];
232 	   wfi[j+4] = ind[i+2];
233 	   wfi[j+5] = ind[i];
234 	   j = j+6;
235 	}
236 	this.wireframe = wfi;
237 };

240 /**
241  * Calculate the bounding box of this model and its centre
242  * 
243  */
244 vxlModel.prototype.computeBoundingBox = function(){	
245     
246     //This is the case with the scene toys
247     if (this.vertices.length == 0){
248         this.bb = [0,0,0,0,0,0];
249         this.centre = [0,0,0,0,0,0];
250         return;
251     }
252     
253 	var vs = this.vertices;
254 	var bbm  = [vs[0],vs[1],vs[2],vs[0],vs[1],vs[2]];
255 	
256 	var i = vs.length;  
257 	for(var i=0, N = vs.length;i<N;i=i+3){
258 		bbm[0] = Math.min(bbm[0],vs[i]);
259 		bbm[1] = Math.min(bbm[1],vs[i+1]);
260 		bbm[2] = Math.min(bbm[2],vs[i+2]);
261 		bbm[3] = Math.max(bbm[3],vs[i]);
262 		bbm[4] = Math.max(bbm[4],vs[i+1]);
263 		bbm[5] = Math.max(bbm[5],vs[i+2]);
264 	}
265 	
266 	
267 	var c = [0, 0, 0];
268      //computes the centre 
269     c[0] = (bbm[3] + bbm[0]) /2;
270     c[1] = (bbm[4] + bbm[1]) /2;
271     c[2] = (bbm[5] + bbm[2]) /2;
272     
273     
274     this.bb = bbm;  
275     this.centre = c;
276 };

278 /**
279  *Returns the bounding box vertices. This method is used by the rendering engine 
280  */
281 vxlModel.prototype.getBoundingBoxVertices = function(){
282     var b = this.bb;
283     return [
284         b[0], b[1], b[2],
285         b[0], b[4], b[2],
286         b[3], b[4], b[2],
287         b[3], b[1], b[2],
288         b[0], b[1], b[5],
289         b[0], b[4], b[5],
290         b[3], b[4], b[5],
291         b[3], b[1], b[5] 
292         ];
293 };
