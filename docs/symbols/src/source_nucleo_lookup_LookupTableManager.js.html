1 /**
  2  * Manages the lookup table files. The constructor will try to load all
  3  * the lookup tables defined in vxl.def.luts at once.
  4  * 
  5  * @class
  6  * @constructor
  7  */
  8 function vxlLookupTableManager(){
  9 	this.lutTimerID = 0;
 10 	this.tables = [];
 11 	this.location = "";
 12 	vxl.go.notifier.addSource(vxl.events.DEFAULT_LUT_LOADED,this);
 13 };
 14 
 15 vxlLookupTableManager.prototype.setLocation = function(loc){
 16 	this.location = loc;
 17 }
 18 
 19 /**
 20  * Load a lookup table file
 21  * @param {String} name the filename of the lookup table to load
 22  */
 23 vxlLookupTableManager.prototype.load = function(name){
 24 		var self = this;
 25 		if (this.isLoaded(name)) return;
 26 
 27 	    var request = new XMLHttpRequest();
 28 	    request.open("GET", this.location+'/'+name+'.lut');
 29 	    request.onreadystatechange = function() {
 30 	      if (request.readyState == 4) {
 31 		    if(request.status == 404) {
 32 				alert (name + ' does not exist');
 33 				vxl.go.console('LookupTableManager: '+name + ' does not exist');
 34 			 }
 35 			else {
 36 				self.handle(name,JSON.parse(request.responseText));
 37 			}
 38 		  }
 39 	    };
 40 		request.send();
 41 };
 42 /**
 43  * Once the lookup table file is retrieved, this method adds it to the lookup table manager
 44  */
 45 vxlLookupTableManager.prototype.handle = function (ID, payload) {
 46 	var lut = new vxlLookupTable();
 47 	lut.load(ID,payload);
 48 	this.tables.push(lut);
 49 	
 50 	if (lut.ID == vxl.def.lut.main){
 51 		vxl.go.notifier.fire(vxl.events.DEFAULT_LUT_LOADED);
 52 	}
 53 };
 54 /**
 55  * Check if a lookup table has been loaded by this lookup table manager
 56  * @param {String} ID the id of the table to check
 57  */
 58 vxlLookupTableManager.prototype.isLoaded = function(ID){
 59 	for(var i=0;i<this.tables.length;i++){
 60 		if (this.tables[i].ID == ID) return true;
 61 	}
 62 	return false;
 63 };
 64 
 65 /**
 66  * Retrieves a lookup table
 67  * @param {String} ID id of the lookup table to retrieve
 68  */
 69 vxlLookupTableManager.prototype.get = function(ID){
 70 	for(var i=0;i<this.tables.length;i++){
 71 		if (this.tables[i].ID == ID) return this.tables[i];
 72 	}
 73 	return null;
 74 };
 75 
 76 /**
 77  * Returns a list with the names of all of the lookup tables that have been loaded.
 78  * @returns {Array} an array with the names of the lookup tables that have been loaded
 79  */
 80 vxlLookupTableManager.prototype.getAllLoaded = function(){
 81     var tablenames = [];
 82     for(var i=0;i<this.tables.length;i++){
 83         tablenames[i] = this.tables[i].ID;
 84     }
 85     return tablenames;
 86 };
 87 
 88 /**
 89  * Checks if all the lookup tables have been loaded
 90  */
 91 vxlLookupTableManager.prototype.allLoaded = function(){
 92 	//@TODO: think of a timeout to alter this state in the case not all tables are loaded (can this happen?)
 93 	return (vxl.def.lut.list.length == this.tables.length);
 94 };
 95 
 96 /**
 97  * Loads all the lookup tables defined in vxl.def.luts
 98  */
 99 vxlLookupTableManager.prototype.loadAll = function(){
100 	for(var i=0;i<vxl.def.lut.list.length;i++){
101 		this.load(vxl.def.lut.list[i]);
102 	}
103 };
104 
105 /**
106  * Creates the global lookup table manager and load all the lookup tables at once
107  */
108 vxl.go.lookupTableManager = new vxlLookupTableManager();
