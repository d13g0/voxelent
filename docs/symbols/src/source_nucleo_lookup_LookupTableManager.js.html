1 /**
  2  * Manages the lookup table files. The constructor will try to load all
  3  * the lookup tables defined in vxl.def.luts at once.
  4  * 
  5  * @class Manages the lookup tables
  6  * @constructor
  7  */
  8 function vxlLookupTableManager(){
  9 	this._hashmap = {};
 10 	this._location = "";
 11 	vxl.go.notifier.publish(vxl.events.DEFAULT_LUT_LOADED,this);
 12 };
 13 
 14 /**
 15  * Relative path to the webpage where lookup tables can be located.
 16  * @param {String} loc
 17  */
 18 vxlLookupTableManager.prototype.setLocation = function(loc){
 19 	this._location = loc;
 20 };
 21 
 22 /**
 23  * Load a lookup table file
 24  * @param {String} name the filename of the lookup table to load
 25  */
 26 vxlLookupTableManager.prototype.load = function(name){
 27     if (this.isLoaded(name)) return;
 28 
 29 	var manager    = this;
 30 	var uri        =  this._location+'/'+name+'.lut';
 31 	var nocacheuri = uri +'?nocache=' + (new Date()).getTime();
 32 
 33 		
 34     var successHandler = function(manager,name){
 35         return function(payload, textStatus){
 36             manager._handle(name,payload);
 37         };
 38    };
 39    
 40    var errorHandler = function(uri){
 41        return function(request, status, error){
 42            if(error.code = 1012){
 43                alert('The file '+uri+' could not be accessed. \n\n'+
 44                'Please make sure that the path is correct and that you have the right pemissions');
 45            }
 46            else{
 47                alert ('There was a problem loading the file '+uri+'. HTTP error code:'+request.status);
 48            }       
 49         };
 50     };
 51 	
 52     var request  = $.ajax({
 53         url         : nocacheuri,
 54         type        :"GET",
 55         dataType    : "json",
 56         success     : successHandler(manager,name),
 57         error       : errorHandler(uri)
 58     }); 
 59 };
 60 
 61 /**
 62  * Once the lookup table file is retrieved, this method adds it to the lookup table manager
 63  */
 64 vxlLookupTableManager.prototype._handle = function (ID, payload) {
 65 	var lut = new vxlLookupTable();
 66 	lut.load(ID,payload);
 67 	this._hashmap[ID] = lut;
 68 	
 69 	if (lut.ID == vxl.def.lut.main){
 70 		vxl.go.notifier.fire(vxl.events.DEFAULT_LUT_LOADED, this);
 71 	}
 72 };
 73 /**
 74  * Check if a lookup table has been loaded by this lookup table manager
 75  * @param {String} ID the id of the table to check
 76  */
 77 vxlLookupTableManager.prototype.isLoaded = function(ID){
 78 	return this._hashmap[ID] != undefined;
 79 };
 80 
 81 /**
 82  * Retrieves a lookup table
 83  * @param {String} ID id of the lookup table to retrieve
 84  */
 85 vxlLookupTableManager.prototype.get = function(ID){
 86 	return this._hashmap[ID];
 87 };
 88 
 89 /**
 90  * Returns a list with the names of all of the lookup tables that have been loaded.
 91  * @returns {Array} an array with the names of the lookup tables that have been loaded
 92  */
 93 vxlLookupTableManager.prototype.getAllLoaded = function(){
 94     var tables = [];
 95     for(lut in this._hashmap){
 96         tables.push(this._hashmap[lut].ID);
 97     }
 98     return tables;
 99 };
100 
101 /**
102  * Checks if all the lookup tables have been loaded
103  */
104 vxlLookupTableManager.prototype.allLoaded = function(){
105     var size = 0;
106 	for(lut in this._hashmap){
107         size++;
108     }
109 	return (vxl.def.lut.list.length == size);
110 };
111 
112 /**
113  * Loads all the lookup tables defined in vxl.def.luts
114  */
115 vxlLookupTableManager.prototype.loadAll = function(){
116 	for(var i=0;i<vxl.def.lut.list.length;i++){
117 		this.load(vxl.def.lut.list[i]);
118 	}
119 };
120 
121 /**
122  * Creates the global lookup table manager and load all the lookup tables at once
123  */
124 vxl.go.lookupTableManager = new vxlLookupTableManager();
