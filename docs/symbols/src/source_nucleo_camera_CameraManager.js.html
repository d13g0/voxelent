 16 /**
 17  * @class
 18  * @constructor
 19  * @param {vxlView} vw the view
 20  * @see vxlCamera
 21  * @see vxlView
 22  */
 23 function vxlCameraManager(vw){
 24 	this.view = vw;
 25 	this.cameras = [];
 26 	this.active = this.createCamera();
 27 	if (vxl.c.camera == undefined){
 28 	    vxl.c.camera = this.active;
 29 	}
 30 }
 31 
 32 /**
 33  * Resets the camera manager and creates one camera
 34  * @param {vxl.def.camera.type} type the type of amera
 35  */
 36 vxlCameraManager.prototype.reset = function(type){
 37 	this.cameras = [];
 38 	this.interactors = [];
 39 	this.active = this.createCamera(type);
 40 };
 41 
 42 /**
 43  * Utilitary method that check if the index idx is between 0 and the size of the camera array
 44  * @param {Number} idx the index to check.
 45  */
 46 vxlCameraManager.prototype.checkBoundary = function(idx){
 47 	if (idx <0 || idx >= this.cameras.length){
 48 		throw('The camera '+idx+' does not exist');
 49 	}
 50 };
 51 /**
 52  * Creates a camera
 53  * @param {vxl.def.camera.type} type the type of camera to create
 54  */
 55 vxlCameraManager.prototype.createCamera = function(type){
 56 	var camera = new vxlCamera(this.view, type);
 57 	
 58 	this.cameras.push(camera);
 59 	camera.idx = this.cameras.length - 1;
 60 	return camera;
 61 };
 62 
 63 /**
 64  * Returns the camera with index idx
 65  * @param {Number} idx the index of the camera to return
 66  */
 67 vxlCameraManager.prototype.getCamera = function(idx){
 68 	this.checkBoundary(idx);
 69 	return this.cameras[idx];
 70 };
 71 
 72 /**
 73  * Returns a reference to the current camera. There is always an active camera
 74  */
 75 vxlCameraManager.prototype.getActiveCamera = function(){
 76 	return this.active;
 77 };
 78 
 79 /**
 80  * Changes the active camera to the camera with index idx
 81  * @param {Number} idx the index of the camera to make active
 82  */
 83 vxlCameraManager.prototype.switchTo = function(idx){
 84     var view = this.view;
 85 	this.checkBoundary(idx);
 86 	this.active = this.cameras[idx];
 87     
 88     if (view.interactor != undefined) {
 89         view.interactor.connectCamera(this.active);
 90     }
 91     else{
 92         vxl.go.console('Camera Manager: switching to camera ['+idx+'] while the view '+view.name+' does not have an interactor set');
 93     }
 94 	return this.active;
 95 };
