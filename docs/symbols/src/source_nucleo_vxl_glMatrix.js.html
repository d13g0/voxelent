426     /**
427      * Projects the specified vec3 from screen space into object space
428      * Based on the <a href="http://webcvs.freedesktop.org/mesa/Mesa/src/glu/mesa/project.c?revision=1.4&view=markup">Mesa gluUnProject implementation</a>
429      *
430      * @param {vec3} vec Screen-space vector to project
431      * @param {mat4} view View matrix
432      * @param {mat4} proj Projection matrix
433      * @param {vec4} viewport Viewport as given to gl.viewport [x, y, width, height]
434      * @param {vec3} [dest] vec3 receiving unprojected result. If not specified result is written to vec
435      *
436      * @returns {vec3} dest if specified, vec otherwise
437      */
438     vec3.unproject = function (vec, view, proj, viewport, dest) {
439         if (!dest) { dest = vec; }
440 
441         if(!unprojectMat) {
442             unprojectMat = mat4.create();
443         }
444 
445         var m = unprojectMat;
446         var v = unprojectVec;
447         
448         v[0] = (vec[0] - viewport[0]) * 2.0 / viewport[2] - 1.0;
449         v[1] = (vec[1] - viewport[1]) * 2.0 / viewport[3] - 1.0;
450         v[2] = 2.0 * vec[2] - 1.0;
451         v[3] = 1.0;
452         
453         mat4.multiply(proj, view, m);
454         if(!mat4.inverse(m)) { return null; }
455         
456         mat4.multiplyVec4(m, v);
457         if(v[3] === 0.0) { return null; }
458 
459         dest[0] = v[0] / v[3];
460         dest[1] = v[1] / v[3];
461         dest[2] = v[2] / v[3];
462         
463         return dest;
464     };

470     /**
471      * Generates a quaternion of rotation between two given normalized vectors
472      *
473      * @param {vec3} a Normalized source vector
474      * @param {vec3} b Normalized target vector
475      * @param {quat4} [dest] quat4 receiving operation result.
476      *
477      * @returns {quat4} dest if specified, a new quat4 otherwise
478      */
479     vec3.rotationTo = function (a, b, dest) {
480         if (!dest) { dest = quat4.create(); }
481 
482         var d = vec3.dot(a, b);
483         var axis = vec3.create();
484         if (d >= 1.0) {
485             quat4.set(identityQuat4, dest);
486         } else if (d < (0.000001 - 1.0)) {
487             vec3.cross(xUnitVec3, a, axis);
488             if (axis.length < 0.000001)
489                 vec3.cross(yUnitVec3, a, axis);
490             if (axis.length < 0.000001)
491                 vec3.cross(zUnitVec3, a, axis);
492             vec3.normalize(axis);
493             quat4.fromAxisAngle(axis, Math.PI, dest);
494         } else {
495             var s = Math.sqrt((1.0 + d) * 2.0);
496             var sInv = 1.0 / s;
497             vec3.cross(a, b, axis);
498             dest[0] = axis[0] * sInv;
499             dest[1] = axis[1] * sInv;
500             dest[2] = axis[2] * sInv;
501             dest[3] = s * 0.5;
502             quat4.normalize(dest);
503         }
504         if (dest[3] > 1.0) dest[3] = 1.0;
505         else if (dest[3] < -1.0) dest[3] = -1.0;
506         return dest;
507     };

604     /**
605      * Calculates the inverse matrix of a mat3
606      *
607      * @param {mat3} mat mat3 to calculate inverse of
608      * @param {mat3} [dest] mat3 receiving inverse matrix. If not specified result is written to mat
609      *
610      * @param {mat3} dest is specified, mat otherwise, null if matrix cannot be inverted
611      */
612     mat3.inverse = function (mat, dest) {
613         var a00 = mat[0], a01 = mat[1], a02 = mat[2],
614             a10 = mat[3], a11 = mat[4], a12 = mat[5],
615             a20 = mat[6], a21 = mat[7], a22 = mat[8],
616 
617             b01 = a22 * a11 - a12 * a21,
618             b11 = -a22 * a10 + a12 * a20,
619             b21 = a21 * a10 - a11 * a20,
620 
621             d = a00 * b01 + a01 * b11 + a02 * b21,
622             id;
623 
624         if (!d) { return null; }
625         id = 1 / d;
626 
627         if (!dest) { dest = mat3.create(); }
628 
629         dest[0] = b01 * id;
630         dest[1] = (-a22 * a01 + a02 * a21) * id;
631         dest[2] = (a12 * a01 - a02 * a11) * id;
632         dest[3] = b11 * id;
633         dest[4] = (a22 * a00 - a02 * a20) * id;
634         dest[5] = (-a12 * a00 + a02 * a10) * id;
635         dest[6] = b21 * id;
636         dest[7] = (-a21 * a00 + a01 * a20) * id;
637         dest[8] = (a11 * a00 - a01 * a10) * id;
638         return dest;
639     };

1392     /**

1394      * If rotating around a primary axis (X,Y,Z) one of the specialized rotation functions should be used instead for performance
1395      *
1396      * @param {mat4} mat mat4 to rotate
1397      * @param {number} angle Angle (in radians) to rotate

1400      *
1401      * @returns {mat4} dest if specified, mat otherwise
1402      */
1403     mat4.rotate = function (mat, angle, axis, dest) {
1404         var x = axis[0], y = axis[1], z = axis[2],
1405             len = Math.sqrt(x * x + y * y + z * z),
1406             s, c, t,
1407             a00, a01, a02, a03,
1408             a10, a11, a12, a13,
1409             a20, a21, a22, a23,
1410             b00, b01, b02,
1411             b10, b11, b12,
1412             b20, b21, b22;
1413 
1414         if (!len) { return null; }
1415         if (len !== 1) {
1416             len = 1 / len;
1417             x *= len;
1418             y *= len;
1419             z *= len;
1420         }
1421 
1422         s = Math.sin(angle);
1423         c = Math.cos(angle);
1424         t = 1 - c;
1425 
1426         a00 = mat[0]; a01 = mat[1]; a02 = mat[2]; a03 = mat[3];
1427         a10 = mat[4]; a11 = mat[5]; a12 = mat[6]; a13 = mat[7];
1428         a20 = mat[8]; a21 = mat[9]; a22 = mat[10]; a23 = mat[11];
1429 
1430         // Construct the elements of the rotation matrix
1431         b00 = x * x * t + c; b01 = y * x * t + z * s; b02 = z * x * t - y * s;
1432         b10 = x * y * t - z * s; b11 = y * y * t + c; b12 = z * y * t + x * s;

1436             dest = mat;

1440             dest[14] = mat[14];
1441             dest[15] = mat[15];
1442         }
1443

1447         dest[2] = a02 * b00 + a12 * b01 + a22 * b02;

1698     /**
1699      * Generates a look-at matrix with the given eye position, focal point, and up axis
1700      *

1702      * @param {vec3} center Point the viewer is looking at
1703      * @param {vec3} up vec3 pointing "up"

1707      */
1708     mat4.lookAt = function (eye, center, up, dest) {
1709         if (!dest) { dest = mat4.create(); }
1710 
1711         var x0, x1, x2, y0, y1, y2, z0, z1, z2, len,
1712             eyex = eye[0],
1713             eyey = eye[1],
1714             eyez = eye[2],
1715             upx = up[0],
1716             upy = up[1],
1717             upz = up[2],
1718             centerx = center[0],
1719             centery = center[1],
1720             centerz = center[2];
1721 
1722         if (eyex === centerx && eyey === centery && eyez === centerz) {
1723             return mat4.identity(dest);
1724         }
1725 
1726         //vec3.direction(eye, center, z);
1727         z0 = eyex - centerx;
1728         z1 = eyey - centery;
1729         z2 = eyez - centerz;
1730 
1731         // normalize (no check needed for 0 because of early return)
1732         len = 1 / Math.sqrt(z0 * z0 + z1 * z1 + z2 * z2);
1733         z0 *= len;
1734         z1 *= len;
1735         z2 *= len;
1736 
1737         //vec3.normalize(vec3.cross(up, z, x));
1738         x0 = upy * z2 - upz * z1;
1739         x1 = upz * z0 - upx * z2;
1740         x2 = upx * z1 - upy * z0;
1741         len = Math.sqrt(x0 * x0 + x1 * x1 + x2 * x2);
1742         if (!len) {
1743             x0 = 0;
1744             x1 = 0;
1745             x2 = 0;
1746         } else {
1747             len = 1 / len;
1748             x0 *= len;
1749             x1 *= len;
1750             x2 *= len;
1751         }
1752 
1753         //vec3.normalize(vec3.cross(z, x, y));
1754         y0 = z1 * x2 - z2 * x1;
1755         y1 = z2 * x0 - z0 * x2;
1756         y2 = z0 * x1 - z1 * x0;
1757 
1758         len = Math.sqrt(y0 * y0 + y1 * y1 + y2 * y2);
1759         if (!len) {
1760             y0 = 0;
1761             y1 = 0;
1762             y2 = 0;
1763         } else {
1764             len = 1 / len;
1765             y0 *= len;
1766             y1 *= len;
1767             y2 *= len;
1768         }
1769 
1770         dest[0] = x0;
1771         dest[1] = y0;
1772         dest[2] = z0;
1773         dest[3] = 0;
1774         dest[4] = x1;
1775         dest[5] = y1;
1776         dest[6] = z1;
1777         dest[7] = 0;
1778         dest[8] = x2;
1779         dest[9] = y2;
1780         dest[10] = z2;
1781         dest[11] = 0;
1782         dest[12] = -(x0 * eyex + x1 * eyey + x2 * eyez);
1783         dest[13] = -(y0 * eyex + y1 * eyey + y2 * eyez);
1784         dest[14] = -(z0 * eyex + z1 * eyey + z2 * eyez);
