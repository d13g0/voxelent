 16 /**
 17  * Creates a vxlCameraState object and associates it with a vxlCamera. 
 18  * 
 19  * This association is unique. each camera has one vxlCameraState object. The vxlCameraState object allows to save/retrive 
 20  * camera states for the camera it is associated with during the construction.
 21  * 
 22  * 
 23  * @class 
 24  * @constructor this is the constructor doc
 25  * @param {vxlCamera} camera
 26  * @author Diego Cantor
 27  * @see vxlCamera
 28   */
 29 
 30 function vxlCameraState(camera) {
 31 	if(!( camera instanceof vxlCamera)) {
 32 		alert('vxlCameraState needs a vxlCamera as argument');
 33 		return null;
 34 	}
 35 
 36 	this.c = camera;
 37 	this.position = new vxlVector3(0, 0, 1);
 38 	this.focalPoint = new vxlVector3(0, 0, 0);
 39 	
 40     this.up = new vxlVector3(0, 1, 0);
 41 	this.right = new vxlVector3(1, 0, 0);
 42     
 43 	this.distance = 0;
 44 	this.azimuth = 0;
 45 	this.elevation = 0;
 46     
 47 	this.xTr = 0;
 48 	this.yTr = 0;
 49 };
 50 
 51 /**
 52  * Resets current camera to the standard location an orientation. This is,
 53  * the camera is looking at the center of the scene, located at [0,0,1] along the z axis and
 54  * aligned with the y axis.
 55  * 
 56  */
 57 vxlCameraState.prototype.reset = function() {
 58 	var c = this.c;
 59 	c.focalPoint = new vxlVector3(0, 0, 0);
 60 	c.up = new vxlVector3(0, 1, 0);
 61 	c.right = new vxlVector3(1, 0, 0);
 62 	c.distance = 0;
 63 	c.elevation = 0;
 64 	c.azimuth = 0;
 65 	c.xTr = 0;
 66 	c.yTr = 0;
 67 	c.setPosition(0, 0, 1);
 68 	c.setOptimalDistance();
 69 };
 70 
 71 /**
 72  * Saves the current state of the camera that this vxlCameraState object is associated with.
 73  */
 74 
 75 vxlCameraState.prototype.save = function() {
 76 	var c = this.c;
 77 	this.distance = c.distance;
 78 	this.azimuth = c.azimuth;
 79 	this.elevation = c.elevation;
 80 	this.xTr = c.xTr;
 81 	this.yTr = c.yTr;
 82 	vxl.vec3.set(c.position, this.position);
 83 	vxl.vec3.set(c.focalPoint, this.focalPoint);
 84 	vxl.vec3.set(c.up, this.up);
 85 	vxl.vec3.set(c.right, this.right);
 86 };
 87 
 88 /**
 89  * Updates the camera with the state stored in vxlCameraState.
 90  */
 91 vxlCameraState.prototype.retrieve = function() {
 92 	var c = this.c;
 93 	c.azimuth = this.azimuth;
 94 	c.elevation = this.elevation;
 95 	c.xTr = this.xTr;
 96 	c.yTr = this.yTr;
 97 
 98 	vxl.vec3.set(this.focalPoint, c.focalPoint);
 99 	vxl.vec3.set(this.up, c.up);
100 	vxl.vec3.set(this.right, c.right);
101 
102 	c.setPosition(this.position.x, this.position.y, this.position.z);
103 };
