 17 /**
 18  * 
 19  * A vxlCamera object simplifies WebGL programming by providing a simple object interface to the lower level
 20  * matrix manipulations that are required to view a 3D scene.
 21  * 
 22  * When moving and rotating the camera, such matrices are updated and Voxelent's Nucleo will use this information
 23  * in order to draw the scene accordingly to the camera position an orientation.
 24  * 
 25  * A vxlCamera object requires a vxlView to be created. Having said that, a vxlView can be associated with 
 26  * multiple cameras.
 27  * 
 28  * The azimuth, elevation and roll operations occur with respect to the standard coordinate system:
 29  * right =[1,0,0], up = [0,1,0] and normal 
 30  * of 
 31  * 
 32  * @class Like a real camera it moves around the 3D scene displaying the current point of view in the browser
 33  * @constructor Creates a vxlCamera. 
 34  * @param {vxlView} vw
 35  * @param {Object} t the type of camera 
 36  * @author Diego Cantor
 37  */
 38 function vxlCamera(vw,t) {
 39     this.UID            = vxl.util.generateUID(); //unique identification key
 40     this.view           = vw;
 41 
 42     this._fov           = vxl.def.camera.fov;
 43     this.Z_NEAR         = vxl.def.camera.near;    
 44     this.Z_FAR          = vxl.def.camera.far;
 45     
 46     this._matrix 	      = mat4.identity();
 47     this._right 		  = vec3.createFrom(1, 0, 0);
 48 	this._up              = vec3.createFrom(0, 1, 0);
 49 	this._forward          = vec3.createFrom(0, 0, 1);	
 50 	this._position        = vec3.createFrom(0, 0, 1);
 51 	this._focalPoint      = vec3.createFrom(0, 0, 0);
 52 	this._distanceVector  = vec3.createFrom(0, 0, 0);
 53     
 54     this._azimuth 	    = 0;
 55     this._elevation     = 0;
 56     this._roll          = 0;
 57 	this._relAzimuth    = 0;
 58     this._relElevation  = 0;
 59     this._relRoll       = 0;
 60 	this._dollyingStep  = 0; //dollying step
 61     this._distance      = 1;
 62     
 63     this._following     = undefined;
 64     this._trackingMode  = vxl.def.camera.tracking.DEFAULT;
 65     
 66     this.landmarks         = [];
 67     this._lmarkAnimationID = undefined; //useful to interrupt landmark animations
 68 
 69 	if (t != undefined && t !=null){
 70         this.type = t;
 71     }
 72     else {
 73         this.type = vxl.def.camera.type.EXPLORING;
 74     }
 75     
 76 };

 78 /**
 79  * Establishes the type of camera
 80  * @param {vxl.def.camera.type} type the type of camera
 81  * @param {vxl.def.camera.tracking} trackingMode if the camera is of tracking type, the tracking mode can be set as 
 82  * an optional parameter here.
 83  * @see {vxl.def.camera}
 84  */
 85 vxlCamera.prototype.setType = function(type, trackingMode){
 86     
 87     if (type != vxl.def.camera.type.ORBITING && 
 88         type != vxl.def.camera.type.TRACKING &&
 89         type != vxl.def.camera.type.EXPLORING) {
 90         console.error('vxlCamera.setType ERROR type'+ type +' unknown. Setting camera to EXPLORING type.');
 91         this.type = vxl.def.camera.type.EXPLORING;
 92     }
 93     else {
 94         this.type = type;
 95     }
 96     
 97     if (this.type  == vxl.def.camera.type.TRACKING && trackingMode != undefined){
 98         this.setTrackingMode(trackingMode);
 99     }
100 };

102 /**
103  * Sets the tracking type of this camera when it follows an actor
104  * <p> to set the tracking type of the camera <code>myCamera</code> you should make sure that your camera is of tracking type with:
105  *  <code>myCamera.setType(vxl.def.camera.type.TRACKING)</code>.
106  *  For instance:
107  * </p>
108  *  <pre class='prettyprint'>
109  *  var actor = vxl.api.getActor('cone'); //from the current scene
110  *  var camera = vxl.c.camera;
111  *  camera.setType(vxl.def.camera.type.TRACKING);
112  *  camera.setTrackingMode(vxl.def.camera.tracking.ROTATIONAL);
113  *  camera.follow(actor);
114  * </pre>
115  * <p> a shorter way would be:</p>
116  * <pre class='prettyprint'>
117  *  var actor = vxl.api.getActor('cone'); //from the current scene
118  *  var camera = vxl.c.camera;
119  *  camera.setType(vxl.def.camera.type.TRACKING);
120  *  camera.follow(actor, vxl.def.camera.tracking.ROTATIONAL);
121  * </pre>
122  * @see vxlCamera#follow, vxlCamera#setTrackingMode
123  */
124 vxlCamera.prototype.setTrackingMode = function(mode){
125     if (this.type != vxl.def.camera.type.TRACKING){
126         alert("Impossible to set a tracking mode if the camera is not of tracking type");
127         throw("Impossible to set a tracking mode if the camera is not of tracking type");
128     }
129     if (mode == undefined) return;
130     this._trackingMode = mode;
131 };

134  * Follows a given actor. If this operation is called on an ORBITING camera,
135  * the camera mode will change to be a TRACKING camera.
136  * 
137  * @param {vxlActor} actor actor to track
138  * @param {String} trackingType one of the possible values of <code>vxl.def.camera.tracking</code>

142     this.setType(vxl.def.camera.type.TRACKING, trackingType);
143     if (actor == undefined || actor == null){
144         alert("vxlCamera.follow: Unable to follow undefined/null actor");
145         console.error("vxlCamera.follow: Unable to follow undefined/null actor");
146     }
147     this._following = actor;
148     actor.addTrackingCamera(this);
149 };

159 /**
160  * Updates the camera according to the current tracking mode. This method is a callback used by the 
161  * vxlActor class to notify the camera of any actor transformations. This is applicable of course only when
162  * this camera is following the actor passed as parameter here
163  * @param {vxlActor} actor the actor being followed by this camera
164  */
165 vxlCamera.prototype.updateWithActor = function(actor){
166     if (this._following != actor) return; //fail safe
167     
168     switch(this._trackingMode){
169         case vxl.def.camera.tracking.DEFAULT:
170                 break;
171         case vxl.def.camera.tracking.ROTATIONAL:
172         case vxl.def.camera.tracking.CINEMATIC:
173                 this.setFocalPoint(actor._position);
174                 break;
175         case vxl.def.camera.tracking.TRANSLATIONAL:
176                 this.translate(actor._translation);  
177                 this.setFocalPoint(actor._position);      
178                 break;
179     }
180 };

182  *<p>Sets the camera position in the scene
183  * This method has three parameters x,y,z which represent the coordinates for 
184  * the camera's position.
185  * </p>
186  * <p>
187  * This method takes into account the current focal point. The camera will look at the
188  * focal point after this operation. If you want to move the camera position and the focal point
189  * simultaneously, then use <code>vxlCamera.translate</code>.
190  * </p>
191  *  
192  *  * @param {Number, Array} x the x-coordinate. x can also be an Array [a,b,c] in this case the y and z parameters are discarded.
193  * @param {Number} y the y-coordinate
194  * @param {Number} z the z-coordinate
195  * @see{vxlCamera#translate}
196  */
197 vxlCamera.prototype.setPosition = function(x,y,z) {
198     this._setPosition(x,y,z);
199     this.setFocalPoint(this._focalPoint);
200 };

202 /**
203  * Looks at a given point in space (sets the focal point of this camera).
204  * 
205  * Note: If the camera is doing cinematic tracking, the up vector will be affected.
206  * @param {Number, Array, vec3} x it can be the x coordinate, a 3-dimensional Array or a vec3 (glMatrix)
207  * @param {Number} y if x is a number, then this parameter corresponds to the y-coordinate
208  * @param {Number} z if x is a number, then this parameter corresponds to the z-coordinate
209  */
210 vxlCamera.prototype.setFocalPoint = function(x,y,z) {
211     
212     var up = vec3.create([0,1,0]);
213     //var up = vec3.create(this._up);
214     this._focalPoint = vxl.util.createVec3(x,y,z);
215     
216     if (this._trackingMode == vxl.def.camera.tracking.CINEMATIC){
217         var d  = vec3.subtract(this._focalPoint, this._position, vec3.create());
218         var x  = d[0], y = d[1],  z = d[2],  r  = vec3.length(d);
219         var el =      Math.asin(y/r)  * vxl.def.rad2deg;
220         var az = 90 + Math.atan2(z,x) * vxl.def.rad2deg;
221         var m = mat4.identity();
222         mat4.rotateY(m, az * vxl.def.deg2rad);
223         mat4.rotateX(m, el * vxl.def.deg2rad);
224         up = mat4.multiplyVec3(m, [0,1,0], vec3.create());
225     }
226     
227     mat4.inverse(mat4.lookAt(this._position, this._focalPoint, up), this._matrix);
228     
229     this._getAxes();
230     this._getDistance();
231     this._getAngles();
232 };

234 /**
235  * Position the camera to a given distance from the current focal point
236  * @param {Number} d the distance
237  * @TODO: REVIEW COMMENTED CODE
238  */
239 vxlCamera.prototype.setDistance = function(d) {
240     
241     if(this._distance == d || d <0) {
242         return;
243     }
244 
245     this._distance = d;
246 
247     // Distance should be greater than .0002
248     if(this._distance < 0.0002) {
249         this._distance = 0.0002;
250         console.warn(" vxlCamera.setDistance WARN: Distance is set to minimum (0.0002)");
251     }
252     
253     this._dollyingStep = this._distance / 100;
254     
255     var pos = vec3.create();
256     
257     var d = this._distance;
258     var n = this._forward;
259     var f = this._focalPoint;
260     
261     pos[0] = d*n[0] + f[0];
262     pos[1] = d*n[1] + f[1];
263     pos[2] = d*n[2] + f[2];
264     
265     this._setPosition(pos);
266 };

344 /**
345  * Changes the azimuth and elevation with respect to the current camera axes
346  * @param {Number} azimuth the relative azimuth
347  * @param {Number} elevation the relative elevation
348  * @param {Number} roll the relative roll
349  */
350 vxlCamera.prototype.rotate = function(azimuth,elevation,roll){
351    
352     if (this.type == vxl.def.camera.type.EXPLORING){ 
353         
354         azimuth   = this._getAngle(azimuth);
355         elevation = this._getAngle(elevation);
356         roll      = this._getAngle(roll);
357         var rotX  = quat4.fromAngleAxis(elevation * vxl.def.deg2rad, [1,0,0]);
358         var rotY  = quat4.fromAngleAxis(azimuth   * vxl.def.deg2rad, [0,1,0]);
359         var rotZ  = quat4.fromAngleAxis(roll      * vxl.def.deg2rad, [0,0,1]); 
360         var rotQ = quat4.multiply(rotY, rotX, quat4.create());
361         rotQ = quat4.multiply(rotQ, rotZ, quat4.create());
362         var rotMatrix = quat4.toMat4(rotQ);
363         mat4.translate(this._matrix, [0,0,-this._distance]);
364         mat4.multiply(this._matrix, rotMatrix);
365         mat4.translate(this._matrix, [0,0,this._distance]);
366     }
367     else {
368         if (Math.abs(this._elevation + elevation) > 90) return; //don't allow
369         this._relElevation = this._getAngle(elevation);
370         this._relAzimuth = this._getAngle(azimuth);
371         this._relRoll = this._getAngle(roll);
372         
373         if (this.type == vxl.def.camera.type.TRACKING){
374             //this._relElevation = -this._relElevation;
375             //this._relAzimuth   = -this._relAzimuth;
376             //this._relRoll      = -this._relRoll;
377         }
378         
379         this._elevation += this._relElevation;
380         this._azimuth += this._relAzimuth;
381         this._roll += this._relRoll;
382         
383          this._computeMatrix();
384     }
385     
386        
387     this._getAxes();
388     if (this.type == vxl.def.camera.type.ORBITING || this.type == vxl.def.camera.type.EXPLORING){
389         this._getPosition();
390     }
391     else if (this.type == vxl.def.camera.type.TRACKING){
392         this._getFocalPoint();
393     }
394     
395    this._update();
396 };

399 /**
400  * Performs the dollying operation in the direction indicated by the camera normal axis.
401  * The dollying mechanism offered by a camera makes sure that the camera moves fast
402  * towards the object when the distance is large and slow when it is very close to the object.
403  * For that effect, every time that the new position  (after dollying) is calculated, the field dstep is computed.
404  * 
405  * @param {Number} value the dollying value 
406  * 
407  */
408 vxlCamera.prototype.dolly = function(value) {
409     
410 	var    n =  this._forward; 
411     var pos = vec3.create(this._position);
412     var step = value*this._dollyingStep;
413     pos[0] += step*n[0];
414     pos[1] += step*n[1];
415     pos[2] += step*n[2];
416     
417     this._setPosition(pos);
418     if (this.type == vxl.def.camera.type.ORBITING || this.type == vxl.def.camera.type.DYNAMIC){
419         this._getDistance();
420     }
421     else if (this.type == vxl.def.camera.type.TRACKING){
422         //move the focal point and keep the distance
423         vec3.add(pos, this._distanceVector, this._focalPoint);
424     }
425 };

565 /**
566  * Retrieves the landmark by name from the known landmarks
567  * @param {String} name the landmark name
568  * @param {Boolean} iflag (optional) interpolation flag. If true it applies a lerp between the
569  * @param {Number} length (optional) the duration of the animation 
570  * @param {Number} fps (optional) the number of frames per second (estmate)  
571  * two landmarks.
572  * @see vxlLandmark
573  */
574 vxlCamera.prototype.gotoLandmark = function(name,iflag,length,fps) {
575 	
576 	var lmark = undefined;
577 	
578 	for(var i=0, N =this.landmarks.length; i<N;i+=1){
579         if (this.landmarks[i].name == name){
580         	lmark = this.landmarks[i];
581         	break;
582         }
583 	}
584 	
585 	if (lmark == undefined){
586 		console.warn('vxlCamera.goTo: landmark with name: '+name+', was not found');
587 		return;
588 	}
589     
590 	if (iflag == undefined || !iflag){
591 		lmark.retrieve(this);
592         return;
593 	}
594 	
595 	if (this._lmarkAnimationID != undefined){
596 		window.clearTimeout(this._lmarkAnimationID);
597 	}
598 	
599 	var self = this;
600 	var dest_pos = lmark._position;
601 	var dest_fp = lmark._focalPoint;
602 	
603 	if (length == undefined){ length = 1000; }
604 	if (fps == undefined) {fps ==  20; } 
605 	
606 	var interactor = this.view.interactor;
607 	interactor.disconnectFromView(); //do not process events during animation
608 	
609 	function animate(length, resolution){
610 		var steps = (length / 100) * (resolution / 10),
611         speed = length / steps,
612         count = 0,
613         start = new Date().getTime();
614 
615 		function iteration(){
616 			
617 			if(count++ != steps){
618 				percent = count/steps;
619 				var inter_fp = vec3.lerp(self._focalPoint, dest_fp,percent);
620 				var inter_pos = vec3.lerp(self._position, dest_pos,percent);
621 				
622 				
623 				self.setFocalPoint(inter_fp);
624 				self.setPosition(inter_pos);
625 				self.refresh();
626 				
627 				var dist = vec3.dist(inter_fp,dest_fp) + vec3.dist(inter_pos,dest_pos);
628 				if (dist >0.01){
629 					var diff = (new Date().getTime() - start) - (count * speed);
630 					self._lmarkAnimationID = window.setTimeout(iteration, (speed - diff));
631 				}
632 				else{
633 					self.setFocalPoint(dest_fp);
634 					self.setPosition(dest_pos);
635 					self.refresh();
636 					interactor.connectView(this.view); //reconnect interactor
637 				}
638 				return;
639 			}
640 		};
641 		
642 		//first time invocation
643 		self._lmarkAnimationID = window.setTimeout(iteration, speed);
644 	}
645 	
646 	animate(length,fps); 
647 	
648 };

667  * This method sets the camera to a distance such that the area covered by the bounding box (parameter)
668  * is viewed.
669  * @param {vxlBoundingBox} bb the bounding box
670  */
671 vxlCamera.prototype._shot = function(bb){
672     
673     this.setElevation(0);
674     this.setAzimuth(0);
675     
676 	var maxDim = Math.max(bb[3] - bb[0], bb[4] - bb[1]);
677 	
678 	cc = [0,0,0];
679 
680 	cc[0] = (bb[3] + bb[0]) /2;
681 	cc[1] = (bb[4] + bb[1]) /2;

683 		
684 	cc[0] = Math.round(cc[0]*1000)/1000;
685 	cc[1] = Math.round(cc[1]*1000)/1000;
686 	cc[2] = Math.round(cc[2]*1000)/1000;
687 	
688 	if(maxDim != 0) {

690 		this.setPosition([cc[0], cc[1], cc[2]+ d]);
691 	}
692 	
693 	this.setFocalPoint(cc);

733 /**
734  * Sets the  camera matrix
735  * @private
736  */
737 vxlCamera.prototype._computeMatrix = function(){
738     
739     mat4.identity(this._matrix);
740     
741     var rotX  = quat4.fromAngleAxis(this._elevation * vxl.def.deg2rad, [1,0,0]);
742     var rotY  = quat4.fromAngleAxis(this._azimuth   * vxl.def.deg2rad, [0,1,0]);
743     var rotZ  = quat4.fromAngleAxis(this._roll      * vxl.def.deg2rad, [0,0,1]); 
744     var rotQ = quat4.multiply(rotY, rotX, quat4.create());
745     rotQ = quat4.multiply(rotQ, rotZ, quat4.create());
746     var rotMatrix = quat4.toMat4(rotQ);
747     
748     if (this.type ==  vxl.def.camera.type.ORBITING || this.type == vxl.def.camera.type.EXPLORING){
749         mat4.translate(this._matrix, this._focalPoint);
750         mat4.multiply(this._matrix, rotMatrix);
751         mat4.translate(this._matrix, [0,0,this._distance]);
752     } 
753     else if(this.type ==  vxl.def.camera.type.TRACKING){
754         mat4.translate(this._matrix, this._position);
755         mat4.multiply(this._matrix, rotMatrix);
756     }
757 };

788  * Sets the camera position in the camera matrix
789  * @private
790  */
791 vxlCamera.prototype._setPosition = function(x,y,z){
792     this._position = vxl.util.createVec3(x,y,z);
793     var m = this._matrix;
794     m[12] = this._position[0];
795     m[13] = this._position[1];
796     m[14] = this._position[2];
797     m[15] = 1;
798 };

800 /**
801  * Recalculates axes based on the current matrix
802  * @private
803  */
804 vxlCamera.prototype._getAxes = function(){
805     var m       = this._matrix;
806     vec3.set(mat4.multiplyVec4(m, [1, 0, 0, 0]), this._right);
807     vec3.set(mat4.multiplyVec4(m, [0, 1, 0, 0]), this._up);
808     vec3.set(mat4.multiplyVec4(m, [0, 0, 1, 0]), this._forward);
809     vec3.normalize(this._right);
810     vec3.normalize(this._up);
811     vec3.normalize(this._forward);
812 };

814 /**
815  * Recalculates the position based on the current matrix
816  * Called only when camera is of ORBITING type
817  * @private
818  */
819 vxlCamera.prototype._getPosition = function(){
820     var m       = this._matrix;
821     vec3.set(mat4.multiplyVec4(m, [0, 0, 0, 1]), this._position);
822     this._getDistance();
823 };

825 /**
826  * Called only when camera is of TRACKING type
827  * @private
828  */
829 vxlCamera.prototype._getFocalPoint = function(){
830     mat4.multiplyVec3(mat4.toRotationMat(this._matrix), [0,0,-this._distance], this._distanceVector); 
831     vec3.add(this._position, this._distanceVector, this._focalPoint);
832     this._getDistance();               
833 };

835 /**
836  * Recalculates the distance variables based on the current state
837  * @private
838  */
839 vxlCamera.prototype._getDistance = function(){
840     this._distanceVector = vec3.subtract(this._focalPoint, this._position, vec3.create());
841     this._distance = vec3.length(this._distanceVector);
842     this._dollyingStep = this._distance / 100;
843 };

845 /**
846  * Recalculates euler angles based on the current state
847  * @private
848  */
849 vxlCamera.prototype._getAngles = function(){
850     //Recalculates angles  
851     var x = this._distanceVector[0], y = this._distanceVector[1],  z = this._distanceVector[2];
852     var r = vec3.length(this._distanceVector);
853     this._elevation = Math.asin(y/r)    * vxl.def.rad2deg;
854     this._azimuth   = Math.atan2(-x,-z) * vxl.def.rad2deg;
855 };

864     this._getPosition();
865     this._getDistance();
866     this._getAngles();

869 /**
870  * @private
871  */
872 vxlCamera.prototype._calculateAngles = function(){
873     var rotM = mat4.toMat3(this._matrix);
874     var Q = mat3.toQuat4(rotM);
875     var x = Q[0], y = Q[1], z = Q[2], w=Q[3];
876     
877     var roll  = Math.atan2(2 * (w * x + y * z),1 - 2 * (x * x + y * y)) * vxl.def.rad2deg;
878     var pitch = Math.asin (2 * (w * y - z * y)) * vxl.def.rad2deg;
879     var yaw   = Math.atan2(2 * (w * z + x * y), 1 - 2 * (y *y + z * z)) * vxl.def.rad2deg;
880     
881     console.info(' roll :' + vxl.util.format(roll,2));
882     console.info('pitch :' + vxl.util.format(pitch,2));
883     console.info('  yaw :' + vxl.util.format(yaw,2));
884 };
