1 /**
  2  * @class A Renderable is an intermediary object between an actor and the rendering strategy that allows rendering
  3  * very complex models.
  4  * 
  5  * Think of this scenario. You have a model that has more than 65K number of vertices.
  6  * Yes, it is going to happen eventually.
  7  * 
  8  * In WebGL you can't do more than 65K indices per draw call. This is because the type of the WebGL index array is unsigned short. 
  9  * So, a model with more than 65K indices could not be rendered at once.
 10  * 
 11  * In previous Voxelent versions (<0.89.3) it was required that a complex model was broken down in several
 12  * JSON files. This was a way to make sure that each 'part' did not get so big as to be unable to render it.
 13  * This is, each part had an index array of at most 16k elements.
 14  * 
 15  * For example, the VTK to JSON importer (vtk2json) handles this situation and produces parts with index arrays of length = 65K.
 16  * 
 17  * However, since version 0.89.2, it is possible to create flat shading models and also to do cell based 
 18  * picking. This requires generating models where triangles are not shared (so we can do flat shading).
 19  * Now, if triangles are not shared, then most likely, very complex models are going to surpass the 65K 
 20  * limit for their index array. 
 21  * 
 22  * The renderable object encapsulates a complex model and delivers parts that abide by the 65K index rule.
 23  * 
 24  * The renderables are requested internally by vxlRenderEngine whenever a model of BIG_OBJECT or  MESH types
 25  * need to be rendered.
 26  * 
 27  * 
 28  * 
 29  * @constructor A Renderable is an intermediary object between an actor and the rendering engine that allows rendering
 30  * very complex models.
 31  * 
 32  * @param{vxlActor} model the actor to be decomposed into renderable parts
 33   *  
 34  * @author Diego Cantor
 35  */
 36 
 37 function vxlRenderable(actor){
 38     
 39     
 40     if (actor == undefined){
 41         throw('vxlRenderable: the actor can not be undefined');
 42     }
 43    
 44     
 45     this.actor = actor;
 46     this.parts = [];    
 47     this.update(vxl.def.renderable.task.CREATE);
 48 }
 49 
 50 
 51 /**
 52  *  Updates the renderable based on changes in the underlying model. After updating, the 
 53  *  renderable parts will contain any changes to the geometry, colors or other model attributes
 54  *  @param {Boolean} reslice if true, it will recreate the parts. Otherwise will use the current parts
 55  */
 56 vxlRenderable.prototype.update = function(task){
 57     
 58     if (task == undefined){
 59         throw ("vxlRenderable.update: Please specify a task");
 60     }
 61 
 62   
 63     var model  = this.actor.getRenderableModel();
 64     
 65     if (model == undefined) return;
 66     
 67     switch(model.type){
 68         case vxl.def.model.type.MESH: this._processMesh(model,task); break;
 69         case vxl.def.model.type.BIG_DATA: this._processBigData(model,task); break;
 70     }
 71    
 72 };
 73 
 74 /**
 75  * This methods creates renderable parts from a mesh. The implementation is straight forward
 76  * given that a mesh does not share triangles. This is what we need to do flat shading and therefore
 77  * to perform cell color based picking.
 78  * 
 79  * 
 80  */
 81 vxlRenderable.prototype._processMesh = function(model,task){
 82     
 83   
 84 
 85     var size = vxl.def.model.MAX_NUM_INDICES;
 86     var N = Math.floor(model.indices.length / size), R = model.indices.length % size;
 87     
 88     if (task == vxl.def.renderable.task.CREATE){
 89         
 90         this.parts = [];
 91         
 92         for (var i=0; i<=N; i +=1){
 93             
 94             var part = new vxlModel(model.name+'-renderable-part-'+i);
 95             var startIndex = i*size;
 96             var endIndex = startIndex + size;
 97             var start = i * size * 3;
 98             var end   = start + size *3;
 99             
100             if (i == N) {
101                 startIndex = N*size;
102                 endIndex = startIndex + R;
103                 start = N * size * 3;
104                 end   = start + R *3;
105                 
106                 if (R==0){
107                     break;
108                 }
109             }
110             
111             part.indices = this._reindex(model.indices.slice(startIndex,endIndex));
112             part.vertices = model.vertices.slice(start, end);
113             
114             
115             if (model.normals && model.normals.length >0){ part.normals  = model.normals.slice(start, end);  }
116             if (model.colors  && model.colors.length >0) { part.colors = model.colors.slice(start, end);     }
117             
118             if (model.pickingColors){
119                 part.pickingColors = model.pickingColors.slice(start, end);
120             }
121             
122             part.update();
123       
124             this.parts.push(part);
125         }
126     }
127     
128     
129     
130     else if (task == vxl.def.renderable.task.UPDATE_COLORS){
131         for (var i=0; i<=N; i +=1){
132             
133             var part = this.parts[i];
134             var start = i * size * 3;
135             var end   = start + size *3;
136             
137             if (i == N) {
138 
139                 start = N * size * 3;
140                 end   = start + R *3;
141                 
142                 if (R==0){
143                     break;
144                 }
145             }
146           
147             if (model.colors  && model.colors.length >0) { part.colors = model.colors.slice(start, end);  }
148 
149         }
150     }
151      
152 };
153 
154 /**
155  * Return indices starting from 0
156  * @param {Object} indices
157  */
158 vxlRenderable.prototype._reindex = function(indices){
159     var min = indices.min();
160     var i = indices.length;
161     while(i--){
162         indices[i] = indices[i] - min;
163     }
164     return indices;
165 }

168 /***
169  * Processing big data 
170  * 
171  * The idea here is read chunks of size 65K from the model index array. Then obtain vertex, normal, and color
172  * arrays for each index in the chunk.
173  * After that, the new part index is generated
174  * 
175  * 
176  */
177 vxlRenderable.prototype._processBigData = function(model,reslice){
178     
179     
180     
181     var bigDataIndex = model.indices;
182     
183     var size = vxl.def.model.MAX_NUM_INDICES;
184     
185     var N = Math.floor(model.indices.length / size);
186     var R = model.indices.length % size;
187     
188     var material = this.actor.material;
189     
190     //TODO: Be more clever about the partitioning. The arrays don't need to have 65K in lenght
191     //what the specs means is that the max element in the array index is 65K.
192     //this is because the index range goes up to 65535 as the highest index.
193     
194     for (var i=0; i<=N; i +=1){
195         
196         var part = new vxlModel(model.name+'-renderable-part-'+i);    
197         var indexMap = [], localIndexArray = [], globalIndexArray = [], innerIndex = 0;
198         
199         if (i == N) {
200             if ( R > 0){
201                 globalIndexArray = bigDataIndex.slice(i*size,i*size+R);
202              }
203              else{
204                  break;
205              }
206         }
207         else{
208             globalIndexArray = bigDataIndex.slice(i*size,(i+1)*size);
209         }
210         
211         if (material.colors && material.colors.length>0)  {  part.colors = [];   }   
212         
213         if (model.normals && model.normals.length>0){  part.normals = [];  }
214         
215         if (model.scalars && model.scalars.length>0){  part.scalars = [];  }
216         
217         for(var k=0,K = globalIndexArray.length; k<K; k+=1){
218             //Get an index from the global index
219             var outerIndex  = globalIndexArray[k];
220             
221             //if it has not been processed then process it.
222             //Processing consist into adding data to the respective part arrays
223             if (indexMap[outerIndex] == undefined){
224                 indexMap[outerIndex] = innerIndex;
225                 
226                 vertexInfo = this._getBigDataVertexInfo(outerIndex);
227                 part.vertices.push.apply(part.vertices, vertexInfo.coords);
228                 if (model.normals && model.normals.length>0){
229  
230                     part.normals.push.apply(part.normals, vertexInfo.normal);
231                 }
232                 if (material.colors && material.colors.length>0){
233                     part.colors.push.apply(part.colors, vertexInfo.color);
234                 }
235                 if (model.scalars && model.scalars.length>0){
236                     part.scalars.push(vertexInfo.scalar);
237                 }
238                 innerIndex +=1;
239             }
240             //write the new index to the part index array
241             part.indices.push(indexMap[outerIndex]);
242             
243         }
244         part.update();
245         this.parts.push(part);
246     }   
247 };

249 /**
250  * Unlike meshes, Big Data models do not have picking colors associated
251  * @param {Object} index
252  * @private
253  */
254 vxlRenderable.prototype._getBigDataVertexInfo = function(index){
255     
256     var material = this.actor.material;
257     var model = this.actor.model;
258     
259     var vertexInfo = {};
260     
261     vertexInfo.coords = model.vertices.slice(index*3, index*3+3);
262     
263     if (model.normals){ vertexInfo.normal = model.normals.slice(index*3, index*3+3);}
264     if (material.colors){  vertexInfo.color = material.colors.slice(index*3, index*3+3);}
265     if (model.scalars){  vertexInfo.scalar = model.scalars[index];}
266     
267     return vertexInfo;
268     
269     
270 };
