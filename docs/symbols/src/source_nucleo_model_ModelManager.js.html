 16 /**
 17  * Creates vxlModel objects and assigns them to a vxlScene. 
 18  * 
 19  * vxlModelManager provides methods for loading remote and local models.
 20  * 
 21  * 
 22  * @class Controls the loading and network transfer of models into memory
 23  * @constructor
 24  * @author Diego Cantor
 25  */
 26 function vxlModelManager(){
 27 	this.toLoad = []; 
 28 	this.models = [];
 29 	
 30 	var e = vxl.events;
 31 	vxl.go.notifier.publish([
 32 	       e.MODELS_LOADING,
 33 	       e.MODEL_NEW,
 34 	       e.MODELS_LOADED
 35 	    ],this);
 36 	vxl.go.notifier.subscribe(e.READER_DONE, this);
 37 };
 38 
 39 /**
 40  * Handles voxelent events to which this model manager is subscribed to
 41  * @param {String} event the event that was fired 
 42  * @param {Object} source the object that fired the event 
 43  */
 44 vxlModelManager.prototype.handleEvent = function(event,source){
 45     var reader = source;
 46     var model = reader.getModel();
 47     this.add(model, reader.scene);
 48 };

 51 /**
 52  * Uses a JSON/Ajax mechanism to load models from a Web Server.
 53  * @param {String} uri The path to the file that will be loaded. 
 54  * @param {vxlScene} scene The scene that will create an actor for this model. This parameter is optional.
 55  */  
 56 vxlModelManager.prototype.load = function(uri, scene) {
 57     var manager = this;
 58     
 59     var filename = uri.replace(/^.*[\\\/]/, '');
 60     var modelname = filename.split('.')[0];
 61     var extension = filename.split('.')[1];
 62 	
 63 	var dtype = 'json';
 64 	
 65 	if (manager.isModelLoaded(modelname)) return;
 66 	
 67 	
 68 	if (extension == 'vtk'){
 69 	    dtype ='text';
 70 	}
 71 	else if (extension == 'json'){
 72 	    dtype = 'json';
 73 	}
 74 	else{
 75 	    alert ('vxlManager.load ERROR: Unknown filetype ['+extension+']');
 76 	    throw ('vxlManager.load ERROR: Unknown filetype ['+extension+']');
 77 	}
 78 	   
 79 	nocacheuri = uri +'?nocache=' + (new Date()).getTime();
 80 	
 81 	
 82 	vxl.go.console('ModelManager.load: Requesting '+uri+'...');
 83 	vxl.go.notifier.fire(vxl.events.MODELS_LOADING, this);
 84 	
 85 	var successHandler = function(manager,modelname,scene){
 86 	    switch(extension){
 87 	    case 'json':
 88     		return function(json, textStatus){
 89     		    json.uri = filename;
 90     		    json.path = vxl.util.getPath(uri);
 91     		    json.name = modelname;
 92     			manager.add(json,scene);
 93     		};
 94     		break;
 95     	case 'vtk':
 96 		    return function(data){
 97 		        reader = new vxlVTKReader(scene);
 98 		        reader.readText(modelname, data);
 99 		    };
100 		 }
101 	};
102 	
103 	var errorHandler = function(uri){
104 		return function(request, status, error){
105 			
106 			if(error.code = 1012){
107 				alert('The file '+uri+' could not be accessed. \n\n'+
108     			'Please make sure that the path is correct and that you have the right pemissions');
109 			}
110 			else{
111 				alert ('There was a problem loading the file '+uri+'. HTTP error code:'+request.status);
112 			}		
113 		};
114 	};
115 	
116 	var request  = $.ajax({
117 		url			: nocacheuri,
118 		type		:"GET",
119 		dataType	: dtype,
120 		success 	: successHandler(manager,modelname,scene),
121 		error		: errorHandler(uri)
122 	});    
123 };

125 /**
126  * Loads a list of models and assigns them to a scene
127  * @param {Array} list list of files to load 
128  * @param {vxlScene} scene scene to callback when the models are loaded 
129  */
130 vxlModelManager.prototype.loadList = function(list,scene){
131 	this.toLoad = list.slice(0); 
132 	vxl.go.console('ModelManager.loadList: models to load ->' + this.toLoad.length);
133    	for(var i=0;i<this.toLoad.length; i++){
134 		this.load(this.toLoad[i],scene);
135    	}
136 };

138 /**
139  * 
140  * @param {Object} p_object the JSON object that contains the definition of the model or
141  * an instance of vxlModel. The object must have a 'name' property.
142  * @param {vxlScene} p_scene the scene to be called back when the model is created
143  */
144 vxlModelManager.prototype.add = function(p_object,p_scene){
145 	
146 	var self = this;
147 	var model = undefined;
148 	
149 	if (p_object instanceof vxlModel){
150 	    model = p_object;
151 	}
152 	else{
153 	    model = new vxlModel(p_object.name, p_object);
154 	}
155 	
156 	function scheduled_add(){
157 		
158     	model.loaded = true;
159     	self.models.push(model);
160     	self.toLoad.splice(name,1); //removes it from the pending list if exists
161     	
162     	vxl.go.console('ModelManager: model '+model.name+' created.'); 
163     	
164     	if (p_scene != undefined && p_scene instanceof vxlScene){
165     		vxl.go.console('ModelManager: notifying the scene...');
166     		if (p_scene.loadingMode == vxl.def.model.loadingMode.LIVE){
167     			p_scene.createActor(model);
168     		}
169     		else if (p_scene.loadingMode == vxl.def.model.loadingMode.LATER){
170     			if(self.allLoaded()){
171     				p_scene.createActors(self.models);
172     			}
173     		}
174     		else if (p_scene.loadingMode == vxl.def.model.loadingMode.DETACHED){
175     			//do nothing
176     		}
177     	}
178         
179     	if(self.allLoaded()){ 
180     		vxl.go.notifier.fire(vxl.events.MODELS_LOADED, self);
181     	}
182     	else{
183     	    vxl.go.notifier.fire(vxl.events.MODEL_NEW, self);
184     	}
185 	};
186 	
187 	setTimeout(function(){scheduled_add();},0);
188  };

190 /**
191  * It will delete all of the loaded models
192  */  
193 vxlModelManager.prototype.reset = function(){
194 	this.firstLoadedModel = false;
195 	for (var i=0; i <this.models.length; i++){
196 		this.models[i] = null;
197 	}
198 	
199 	this.models        = [];
200 	this.toLoad        = [];
201 };
202 
203 /**
204  * Checks if a model has been loaded yet
205  * @param {String} name the name of the model to check
206  */
207 vxlModelManager.prototype.isModelLoaded = function(name){
208 	for(var i=0;i<this.models.length;i++){
209 		if (this.models[i].name == name) return true;
210 	}
211 	return false;
212 };
213 
214 /**
215  * Returns true if all the models in the list passed to the method loadList have been loaded.
216  */
217 vxlModelManager.prototype.allLoaded = function(){
218 	return (this.toLoad.length == 0);
219 };

222 /**
223  * Returns the model if it has been loaded by this model manager, null otherwise.
224  * @param {String} name the name of the model to retrieve
225  */
226 vxlModelManager.prototype.getModelByName = function(name){
227  	for(var i=0, max = this.models.length; i<max; i+=1){
228 		if (this.models[i].name == name) return this.models[i];
229 	}
230 	return null;
231 };
