 17 /**
 18 *  Sometimes it makes sense to associate actors in a scene to perform operations as a unit.
 19 *  For instance you can have a bicycle composed by several actors and you want to change 
 20 *  its color or its position on the scene. You could accomplish that calling the respective
 21 *  methods for each one of the actors that make part of the bicycle or you can create an
 22 *  actor group named bycicle and call just the setColor and translate operations on the actor group.
 23 * 
 24 *  @class
 25 *  @constructor
 26 *  @author Diego Cantor
 27 */
 28 function vxlActorGroup(scene, name, list){
 29     this.scene = scene;
 30     this.name = name;
 31     this.list = [];
 32     this.addList(list);
 33 };
 34 
 35 /**
 36  *  Add a list of actors to the actor group
 37  * @param {Object} list a list of actors
 38  */
 39 vxlActorGroup.prototype.addList = function(list){
 40     var messages = [];
 41     
 42     for(var i=0, N = list.length; i <N; i+=1){
 43         var actor = list[i];
 44         if (actor instanceof vxlActor && actor.scene == this.scene){
 45             this.list.push(actor);
 46         }
 47         else if (typeof(actor) == 'string'){
 48             var actorObject  = this.scene.getActor(actor);
 49             if (actorObject != null){
 50                 this.list.push(actorObject);
 51             }
 52             else{
 53                 messages.push("Could not find actor: "+actor);
 54             }
 55         }
 56         else{
 57             messages.push("The object "+actor+" is not of the expected type (vxlActor, string)");
 58         }
 59     }
 60     
 61     if (messages.length != 0){
 62         this.list = [];
 63         throw new vxlActorGroupException(messages);
 64     }
 65     
 66 };
 67 
 68 /**
 69 * Adds one actor to the actor group
 70 * @param {vxlActor} actor the actor to be added
 71 */
 72 vxlActorGroup.prototype.add = function(actor){
 73     var messages = [];
 74 
 75     if (actor instanceof vxlActor && actor.scene == this.scene){
 76         this.list.push(actor);
 77     }
 78     else if (typeof(actor) == 'string'){
 79         var actorObject  = this.scene.getActor(actor);
 80         if (actorObject != null){
 81             this.list.push(actorObject);
 82         }
 83         else{
 84             messages.push("Could not find actor: "+actor);
 85         }
 86     }
 87     else{
 88         messages.push("The object "+actor+" is not of the expected type (vxlActor, string)");
 89     }
 90     
 91     if (messages.length != 0){
 92         this.list = [];
 93         throw new vxlActorGroupException(messages);
 94     }
 95 };
 96 
 97 /**
 98  * Returns true if the actor is associated to this actor group. False otherwise
 99  */
100 vxlActorGroup.prototype.hasActor = function(actor){
101     if (actor instanceof vxlActor){
102         return (this.list.indexOf(actor)!=-1)
103     }
104     else if (typeof(actor)=='string'){
105         var actorObject = this.scene.getActor(actor);
106         return (actorObject != null);
107     }
108     
109     else return false;
110 };
111 
112 vxlActorGroup.prototype.remove = function(actor){
113     var index = -1;
114     if (actor instanceof vxlActor){
115         index = this.list.indexOf(actor);
116         
117     }
118     else if (typeof(actor)=='string'){
119         actorObject = this.scene.getActor(actor);
120         index = this.list.indexOf(actorObject);
121     }   
122    
123     if (index !=1){
124         this.list.splice(index,1);
125     }
126     else{
127         throw new vxlActorGroupException(['the actor '+actor+' was not found in the group '+this.name]);
128     }
129 };

132 /**
133  * Returns the size of the actor group
134  * @returns {Number} the lenght of the actor group
135  */
136 vxlActorGroup.prototype.size = function(){
137     return this.list.length;
138 };
139 
140 /**
141  * Set a property for actors in the group. Please notice that by design actor groups only 
142  * set actor level properties instead of model level properties.
143  * 
144  */
145 vxlActorGroup.prototype.setProperty = function(property, value){
146     for(var i=0, N = this.list.length; i<N; i+=1){
147         this.list[i].setProperty(property,value,vxl.def.actor);
148     }
149     
150     return this;
151 };
152 
153 /**
154  * Invoke an operation on the actors belonging to this group
155  * @param {function} operation from vxlActor.prototype
156  * @param {list} parameters
157  */
158 vxlActorGroup.prototype._apply = function(operation, parameters){
159    for(var i=0, N = this.list.length; i<N; i+=1){
160         operation.apply(this.list[i], parameters);
161     };
162     
163     return this;
164 };
