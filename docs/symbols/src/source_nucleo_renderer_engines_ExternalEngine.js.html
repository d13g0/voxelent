 17 vxlExternalEngine.prototype = new vxlEngine();
 18 vxlExternalEngine.prototype.constructor = vxlExternalEngine;
 19 /**
 20  * 
 21  * @param {Object} allocate
 22  * @param {Object} render
 23  * @param {Object} deallocate
 24  */
 25 function vxlExternalEngine(renderer, allocate, render, deallocate){
 26     vxlEngine.call(this);
 27     this.renderer = renderer;
 28     this.allocateCallback = allocate;
 29     this.renderCallback = render;
 30     this.deallocateCallback = deallocate;
 31 }
 32 
 33 
 34 vxlExternalEngine.prototype.allocate = function(scene){
 35     this.allocateCallback(this.renderer, scene);
 36 };
 37 
 38 
 39 vxlExternalEngine.prototype.render = function(scene){
 40     this.renderCallback(this.renderer, scene);  
 41 };
 42 
 43 vxlExternalEngine.prototype.deallocate = function(scene){
 44     this.deallocateCallback(this.renderer, scene);  
 45 };
