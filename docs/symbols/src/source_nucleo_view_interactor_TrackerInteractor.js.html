 18 vxlTrackerInteractor.prototype = new vxlViewInteractor();
 19 vxlTrackerInteractor.prototype.constructor = vxlViewInteractor;
 20 /**
 21  * @class
 22  * @constructor
 23  * Interprets mouse and keyboard events and translate them to camera actions
 24  * @augments vxlViewInteractor
 25  */
 26 function vxlTrackerInteractor(view,camera){
 27     vxlViewInteractor.call(this, view, camera);    
 28 	this.MOTION_FACTOR = 10.0;
 29 	this.task = vxl.def.camera.task.NONE;
 30 	this.x = 0;
 31 	this.y = 0;
 32 	this.lastX = 0;
 33 	this.lastY = 0;
 34 	this.lastClickedX = 0;
 35 	this.lastClickedY = 0;
 36 	this.ctrlPressed = false;
 37 	this.altPressed = false;
 38 	this.keyPressed = 0;
 39 	this.button = -1;
 40 	this.dragging = false;
 41 };
 42 
 43 /**
 44  * Returns the type of this interactor as a string 
 45  */
 46 vxlTrackerInteractor.prototype.getType = function(){
 47     return "vxlTrackerInteractor";
 48 };
 49 
 50 /**
 51  * Reacts to the window onKeyDown event 
 52  */
 53 vxlTrackerInteractor.prototype.onKeyDown = function(ev){
 54     this.keyPressed = ev.keyCode;
 55     this.altPressed = ev.altKey;
 56     this.shiftPressed = ev.shiftKey;
 57     
 58     var camera = this.camera;
 59     
 60     //ROTATING
 61     if (!this.altPressed && !this.shiftPressed){
 62         switch(this.keyPressed){
 63               case 38:camera.changeElevation(10);  break;
 64               case 40:camera.changeElevation(-10); break;
 65               case 37:camera.changeAzimuth(-10);   break;
 66               case 39:camera.changeAzimuth(10);    break;
 67               default: break;
 68         }
 69     }
 70     //PANNING
 71     else if(this.shiftPressed && this.keyPressed !=17) {
 72         var px = 0;
 73         var py = 0;
 74         switch(this.keyPressed){
 75             case 38:py = 10; break;
 76             case 40:py = -10;break;
 77             case 37:px = -10;break;
 78             case 39:px = 10; break;
 79             default: break;
 80         }
 81         if(px != 0 || py !=0){
 82             this.pan(px,py);
 83         }
 84     }
 85     this.camera.refresh();
 86 };
 87 
 88 /**
 89  * Reacts to the canvas onkeyup event 
 90  */
 91 vxlTrackerInteractor.prototype.onKeyUp = function(ev){
 92     if (ev.keyCode == 17){
 93         this.ctrlPressed = false;
 94     }
 95 };

 97 /**
 98  *Reacts to the canvas onmouseup event 
 99  */
100 vxlTrackerInteractor.prototype.onMouseUp = function(ev){
101 	task = vxl.def.camera.task.NONE;
102 	this.dragging = false;
103 };

105 /**
106  *Reacts to the canvas onmousedown event 
107  */
108 vxlTrackerInteractor.prototype.onMouseDown = function(ev){
109 	this.x             = ev.clientX;
110 	this.y             = ev.clientY;
111 	this.lastClikedX   = this.x;
112 	this.lastClickedY  = this.y;
113 	this.button        = ev.button;
114 	this.dragging      = true;
115 };
116 
117 /**
118  *Reacts to the canvas onmousemove event 
119  */
120 vxlTrackerInteractor.prototype.onMouseMove = function(ev){
121 
122 	this.lastX         = this.x;
123 	this.lastY         = this.y;
124 	this.x             = ev.clientX;
125     this.y             = ev.clientY;
126 
127 	if (!this.dragging) return;
128     if (this.button !=0) return;  
129 	
130 	this.ctrlPressed 	= ev.ctrlKey;
131 	this.altPressed 	= ev.altKey;
132 	this.shiftPressed 	= ev.shiftKey;
133 	
134 	var rx = this.x - this.lastX;
135 	var ry = this.y - this.lastY;
136 	
137 	
138 	 
139     if (this.altPressed){ 
140 		this.dolly(ry);
141 	}
142 	else if (this.shiftPressed){
143 		this.pan(rx,ry);
144 	}
145 	else{
146 	    this.rotate(rx,ry);
147 	}
148 	
149 	this.camera.refresh();
150 };

153 /**
154  * Internal method used by this tracker to perform dollying
155  * @param {Number} value the number of dollying steps
156  */
157 vxlTrackerInteractor.prototype.dolly = function(value){
158 	
159 	this.task = vxl.def.camera.task.DOLLY;
160     this.camera.dolly(value);
161 };
162 
163 /**
164  * Internal method used by this tracker to rotate the camera.
165  * @param {Number} dx the rotation on the X axis (elevation)
166  * @param {Number} dy the rotation on the Y axis (azimuth)
167  */
168 vxlTrackerInteractor.prototype.rotate = function(rx, ry){
169 	
170 	this.task = vxl.def.camera.task.ROTATE;
171 	
172 	var canvas = this.camera.view.canvas;
173 	var dx = -20.0 / canvas.height;
174 	var dy = -20.0 / canvas.width;
175 	var rotX = rx * dx * this.MOTION_FACTOR;
176 	var rotY = ry * dy * this.MOTION_FACTOR;
177 
178 	this.camera.rotate(rotX, rotY);
179 };
180 
181 /**
182  * Internal method used by this tracker to perform panning 
183  * @param {Object} dx
184  * @param {Object} dy
185  */
186 vxlTrackerInteractor.prototype.pan = function(dx,dy){
187 
188 	this.task = vxl.def.camera.task.PAN;
189 	
190 	var camera = this.camera;
191 	var canvas = camera.view.canvas;
192 	var scene = camera.view.scene;
193 	var dimMax = Math.max(canvas.width, canvas.height);
194 	var deltaX = 1 / dimMax;
195 	var deltaY = 1 / dimMax;
196 	var ndx = dx * deltaX * this.MOTION_FACTOR * scene.bb.max();
197 	var ndy = dy * deltaY * this.MOTION_FACTOR * scene.bb.max();
198 
199 	camera.pan(ndx,ndy);
200 };
