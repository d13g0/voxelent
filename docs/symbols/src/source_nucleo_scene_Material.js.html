 18 function vxlMaterial(model){
 19     this.ambient   = vxl.def.material.ambient;
 20     this.diffuse   = vxl.def.material.diffuse;
 21     this.specular  = vxl.def.material.specular;
 22     this.shininess = vxl.def.material.shininess;
 23     this.opacity   = vxl.def.material.opacity;
 24     this.shading   = vxl.def.material.shading;
 25     this.texture   = undefined;
 26     this.colors    = undefined;
 27 
 28     if (model){
 29         this.getFrom(model);
 30     } 
 31 };
 32 
 33 /**
 34  * 
 35  * @param {Object} model the model from where the material properties are read
 36  */
 37 vxlMaterial.prototype.getFrom = function(model){
 38  
 39     
 40     if (model.ambient != undefined)
 41     {
 42         this.ambient = model.ambient.slice(0);
 43     }
 44     
 45     if (model.diffuse != undefined){
 46         this.diffuse = model.diffuse.slice(0) 
 47     }
 48     
 49     if (model.specular != undefined){
 50         this.specular = model.specular.slice(0);
 51     }
 52     
 53     if (model.color != undefined){
 54         this.diffuse = model.color.slice(0);
 55     }
 56     
 57     if (model.shininess != undefined){
 58         this.shininess = model.shininess;
 59     }
 60     
 61     if (model.opacity != undefined){
 62         this.opacity = model.opacity;
 63     }
 64     
 65     if (model.shading != undefined){
 66         this.shading = model.shading;
 67     }
 68     
 69     if (model.texture != undefined){
 70         this.texture = new vxlTexture(model.path + model.texture); 
 71     }
 72     
 73     if (model.colors != undefined){
 74         this.colors = model.colors;
 75     }
 76     
 77     
 78 };
 79 
 80 /**
 81  *  Clones this material
 82  *  @see vxlActor#clone
 83  */
 84 vxlMaterial.prototype.clone = function(){
 85     var copy = new vxlMaterial();
 86     var self = this;
 87      
 88     for (var prop in self){
 89         if (self.hasOwnProperty(prop)) {
 90             if (self[prop] instanceof Array){
 91                 copy[prop] = self[prop].slice(0);
 92             }
 93             else {
 94                copy[prop] = self[prop];               
 95             }
 96         }
 97     }
 98     return copy;
 99 };
