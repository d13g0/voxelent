 21 /**
 22  * @constructor
 23  * @class
 24  * Implements a basic rendering strategy that works with the following programs:
 25  * 
 26  * vxl.def.glsl.bake
 27  */
 28 function vxlBakeStrategy(renderer){
 29     vxlBasicStrategy.call(this,renderer);
 30     
 31     var gl = renderer.gl;
 32     
 33     this._calls     = {}
 34     
 35     this._gl_buffers  = {
 36         index       : gl.createBuffer(),
 37         baked       : gl.createBuffer(),
 38         position    : gl.createBuffer(),
 39         scale       : gl.createBuffer(),
 40         shading     : gl.createBuffer()
 41     };  
 42     
 43     this._data        = {
 44         index       : [],
 45         baked       : [],
 46         position    : [],
 47         scale       : [],
 48         shading     : []
 49         
 50     }; 
 51      
 52     this._allocated   = []; //if an actor is here, do not allocate
 53     
 54     this._offsets     = {
 55        position :{},
 56        scale    :{},
 57        shading  :{},
 58        baked    :{} 
 59     }; 
 60     
 61     
 62     this.glsl = vxl.util.extend(
 63         vxl.def.glsl,{
 64         POSITION: "aPosition",
 65         SCALE:    "aScale",
 66         SHADING:  "aShading"
 67     });
 68     
 69     var e = vxl.events;
 70     vxl.go.notifier.subscribe([
 71         e.ACTOR_MOVED,
 72         e.ACTOR_SCALED,
 73         e.ACTOR_CHANGED_SHADING,
 74         e.ACTOR_CHANGED_COLOR
 75     ], this);
 76 };
 77 
 78 /**
 79  * Handle voxelent events
 80  */
 81 vxlBakeStrategy.prototype.handleEvent = function(event, actor){
 82     
 83     switch(event){
 84         case vxl.events.ACTOR_MOVED:            this._updateActorPosition(actor);   break;
 85         case vxl.events.ACTOR_SCALED:           this._updateActorScale(actor);      break;
 86         case vxl.events.ACTOR_CHANGED_SHADING:  this._updateActorShading(actor);    break;
 87         case vxl.events.ACTOR_CHANGED_COLOR:    this._updateActorColor(actor);      break;
 88     }
 89 };

 93 /**
 94  * Creates an array of a defined size populated with the indicated value
 95  * @private
 96  */
 97 vxlBakeStrategy.prototype._populate = function(value, size){
 98     var a = [];
 99     
100     for (var i = 0; i < size; i+=1){
101         if (value instanceof Array || value instanceof Float32Array){
102             for (var j = 0; j < value.length; j+=1){
103                 a.push(value[j]);
104             }
105         }
106         else{
107             a.push(value);
108         }
109     }
110     return a;
111 };

114 /**
115  * Computes the number of required calls to render one scene
116  * @private
117  */
118 vxlBakeStrategy.prototype._computeRequiredCalls = function(scene){
119 
120     var elements    = scene.actors; //.concat(scene.toys.list);
121     var NUM         = elements.length;
122     var accum = 0;
123     var count = 1;
124     
125     for(var i = 0; i < NUM; i+=1){
126         accum += elements[i].model.indices.length;
127         if (accum > 65000){ //66560?
128             accum = 0;
129             count ++;
130         }
131     }
132     return count;
133 };

136 /**
137  * Receives one actor and returns the GL buffers
138  * @private
139  */
140 vxlBakeStrategy.prototype._allocateActor = function(actor){
141     
142     if (this._allocated.indexOf(actor.UID) != -1) return; //this actor has been allocated
143     var data            = this._data;
144     var offsets         = this._offsets;
145     var gl              = this.renderer.gl;
146     var model           = actor.model;
147     var NUM_VERTICES    = model.vertices.length;
148     
149         
150     var color = [], normal = [];
151     
152     //Taking care of colors
153     if (actor.color) {
154         color = this._populate(actor.color, NUM_VERTICES/3);
155     }
156     else{
157         color = this._populate([0.7,0.7,0.7], NUM_VERTICES/3);
158     }
159     
160     //Taking care of normals
161     if (model.normals){
162         normal = model.normals;
163     }
164     else{
165         normal = this._populate(0, NUM_VERTICES);
166     }
167     
168     offsets.baked[actor.UID]    =data.baked.length;
169     
170     for (var i=0;i<NUM_VERTICES; i+=3){
171         
172         data.baked.push(model.vertices[i]);
173         data.baked.push(model.vertices[i+1]);
174         data.baked.push(model.vertices[i+2]);
175         
176         data.baked.push(normal[i]);
177         data.baked.push(normal[i+1]);
178         data.baked.push(normal[i+2]);
179         
180         data.baked.push(color[i]);
181         data.baked.push(color[i+1]);
182         data.baked.push(color[i+2]);
183     }
184     
185     offsets.position[actor.UID] = data.position.length;
186     offsets.scale[actor.UID]    = data.scale.length;
187     offsets.shading[actor.UID]  = data.shading.length;
188     
189     data.position = data.position.concat(this._populate(actor.position, NUM_VERTICES/3));
190     data.scale    = data.scale.concat(this._populate(actor.scale, NUM_VERTICES/3));
191     
192     if (actor.shading){
193         data.shading = data.shading.concat(this._populate(1.0, NUM_VERTICES/3));
194     }
195     else{
196         data.shading = data.shading.concat(this._populate(0.0, NUM_VERTICES/3));
197     }
198 
199     var ind = model.indices.slice(0);
200     if (data.index.length > 0){
201         var max = data.index.max()+1;
202         var NUM_INDICES = model.indices.length;
203         for (var i=0; i<NUM_INDICES; i+=1){
204             ind[i] += max;
205         }
206         data.index = data.index.concat(ind);
207     }
208     else{
209         data.index = ind;
210     }
211     
212     this._allocated.push(actor.UID);
213 };

215 /**
216  * @private
217  */
218 vxlBakeStrategy.prototype._updateActorPosition = function(actor){
219     
220     var data = this._data;
221     var offset = this._offsets.position[actor.UID];
222     
223     if (offset == undefined) return;
224     
225     var LEN = actor.model.vertices.length + offset;
226     for(var i =offset;i<LEN;i+=3){
227         data.position[i]   = actor.position[0];
228         data.position[i+1] = actor.position[1];
229         data.position[i+2] = actor.position[2];
230     }
231 };
232 
233 /**
234  * @private
235  */
236 vxlBakeStrategy.prototype._updateActorScale = function(actor){
237     var data = this._data;
238     var offset = this._offsets.scale[actor.UID];
239     
240     if (offset == undefined) return;
241     
242     var LEN = actor.model.vertices.length + offset;
243     for(var i =offset;i<LEN;i+=3){
244         data.scale[i]   = actor.scale[0];
245         data.scale[i+1] = actor.scale[1];
246         data.scale[i+2] = actor.scale[2];
247     }
248 };

251 /**
252  * @private
253  */
254 vxlBakeStrategy.prototype._updateActorColor = function(actor){
255     var data = this._data;
256     var offset = this._offsets.baked[actor.UID];
257     
258     if (offset == undefined) return;
259     
260     var LEN = (actor.model.vertices.length*3) + offset;
261     for(var i =offset;i<LEN;i+=9){
262         data.baked[i+6] = actor.color[0];
263         data.baked[i+7] = actor.color[1];
264         data.baked[i+8] = actor.color[2];
265     }
266 };
267 
268 /**
269  * @private
270  */
271 vxlBakeStrategy.prototype._updateActorShading = function(actor){
272     var data = this._data;
273     var offset = this._offsets.shading[actor.UID];
274     
275     if (offset == undefined) return;
276     
277     var LEN = (actor.model.vertices.length/3) + offset;
278     var val = 0.0;
279     
280     if (actor.shading){
281         val = 1.0;
282     }
283     
284     for(var i =offset;i<LEN;i+=1){
285        data.shading[i]   = val;
286     }
287 };

298 /**
299  * Implements basic allocation of memory. Creates the WebGL buffers for the actor
300  * @param {vxlScene} scene the scene to allocate memory for
301  * @returns an object that contains the allocated WebGL buffers
302  */
303 vxlBakeStrategy.prototype.allocate = function(scene){
304     
305     if (this._computeRequiredCalls(scene) > 1) {
306         throw 'Not renderable yet. Working on it. The number of indices exceeds the 65K limit';
307     }
308     
309     var elements    = scene.actors; //.concat(scene.toys.list);
310     var NUM         = elements.length;
311     var r           = this.renderer;
312     var buffers     = this._gl_buffers;
313     var data        = this._data;
314 
315     var prg         = r.prg;
316     var gl          = r.gl;
317     var glsl        = this.glsl;
318     var mode        = gl.STATIC_DRAW;
319     
320     
321     
322     for(var i = 0; i < NUM; i+=1){
323         this._allocateActor(elements[i]);
324     }
325     
326     prg.enableAttribute(glsl.VERTEX_ATTRIBUTE);
327     prg.enableAttribute(glsl.NORMAL_ATTRIBUTE);
328     prg.enableAttribute(glsl.COLOR_ATTRIBUTE);
329     
330     prg.enableAttribute(glsl.POSITION);
331     prg.enableAttribute(glsl.SCALE);
332     prg.enableAttribute(glsl.SHADING);
333     
334     gl.bindBuffer(gl.ARRAY_BUFFER, buffers.baked);
335     gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(data.baked), mode);
336     prg.setAttributePointer(glsl.VERTEX_ATTRIBUTE, 3, gl.FLOAT, false, 36, 0);
337     prg.setAttributePointer(glsl.NORMAL_ATTRIBUTE, 3, gl.FLOAT, false, 36, 12);
338     prg.setAttributePointer(glsl.COLOR_ATTRIBUTE, 3, gl.FLOAT, false, 36, 24);
339     
340     gl.bindBuffer(gl.ARRAY_BUFFER, buffers.position);
341     gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(data.position), mode);
342     prg.setAttributePointer(glsl.POSITION, 3, gl.FLOAT, false,12,0);
343     
344     gl.bindBuffer(gl.ARRAY_BUFFER, buffers.scale);
345     gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(data.scale), mode);
346     prg.setAttributePointer(glsl.SCALE, 3, gl.FLOAT, false,12,0);
347     
348     gl.bindBuffer(gl.ARRAY_BUFFER, buffers.shading);
349     gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(data.shading), mode);
350     prg.setAttributePointer(glsl.SHADING, 1, gl.FLOAT, false,4,0);
351     
352 
353     gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, buffers.index);
354     gl.bufferData(gl.ELEMENT_ARRAY_BUFFER, new Uint16Array(data.index), mode);
355 };

360 /**
361  * Renders the actors one by one
362  * @param {vxlScene} scene the scene to render
363  */
364 vxlBakeStrategy.prototype.render = function(scene){
365 
366     //Updates the perspective matrix and passes it to the program
367     var r       = this.renderer;
368     var trx     = r.transforms
369     var prg     = r.prg;
370     var gl      = r.gl;
371     var glsl    = vxl.def.glsl;
372     var data    = this._data;
373     
374     trx.update();
375 
376     prg.setUniform(glsl.PERSPECTIVE_MATRIX, trx.pMatrix);
377     prg.setUniform(glsl.MODEL_VIEW_MATRIX,  trx.mvMatrix);
378     prg.setUniform(glsl.NORMAL_MATRIX,      trx.nMatrix);
379     prg.setUniform(glsl.MVP_MATRIX,         trx.mvpMatrix);
380     
381     gl.drawElements(gl.TRIANGLES, data.index.length, gl.UNSIGNED_SHORT,0);
382     gl.bindBuffer(gl.ARRAY_BUFFER, null);
383     gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, null);
384 };
