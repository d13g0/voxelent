 18 /**
 19  * <p>
 20  * A vxlTexture is a representation of a raster image in Voxelent. Textures can be loaded providing a
 21  * URI
 22  * </p>
 23  * <p> 
 24  * To set the magnification and minification filters for a texture please refer to the constants
 25  * defined in <code>vxl.def.texture.filter</code>. A vxlTexture object uses the <code>min</code> and
 26  * <code>mag</code> properties to set these filters. For example: </p>
 27  * 
 28  * <pre class="prettyprint">
 29  * var texture = new vxlTexture('iphone_screen.png');
 30  * texture.min = vxl.def.texture.filter.LINEAR;
 31  * texture.mag = vxl.def.texture.filter.NEAREST;
 32  * </pre>
 33  * 
 34  * <p>The maginfication and minification filters by default are:
 35  * vxl.def.texture.filter.LINEAR and vxl.def.texture.filter.LINEAR_MIPMAP_LINEAR respectively    
 36  * </p>
 37  * 
 38  * <p>Under normal circumstances you will not need to create a vxlTexture. An actor representing a model with texture
 39  * information in it will create a vxlTexture automatically. In this case you can access the available vxlTexture object from the actor like this:</p>
 40  *  
 41  * <pre class="prettyprint">
 42  * var actor = vxl.c.scene.getActorByName('iphone_screen.json');
 43  * actor.texture.min = [set the filter here using the constants defined in vxl.def.texture.filter]
 44  * actor.texture.mag = [set the filter here using the constants defined in vxl.def.texture.filter]
 45  * </pre>
 46  * 
 47  * <p>If you want to replace the texture object with a new raster image, you can write something like this:</p>
 48  * 
 49  * <pre class="prettyprint">
 50  * var actor = vxl.c.scene.getActorByName('iphone_screen.json');
 51  * var wallpaper = new vxlTexture('new_wallpaper.png');
 52  * actor.setTexture(wallpaper);
 53  * </pre>
 54  * @class A vxlTexture is a representation of a raster image in Voxelent. 
 55  * @constructor
 56  * @param {String} uri texture location
 57  */
 58 function vxlTexture(uri){
 59     var self = this;
 60 
 61     this.image = new Image();
 62     this.image.onload = function(){
 63         self._handleLoadedImage();
 64     }
 65     
 66     this.uri = uri;
 67     if (this.uri != undefined){
 68         this.load(this.uri);
 69     }
 70     
 71     this.mag = vxl.def.texture.filter.LINEAR;
 72     this.min = vxl.def.texture.filter.LINEAR_MIPMAP_LINEAR;     
 73 };
 74 
 75 /**
 76  * Loads an image and it associates it to this texture object
 77  * @param {Object} uri the location of the image to load into this texture object
 78  */
 79 vxlTexture.prototype.load = function(uri){
 80    this.image.src = uri;
 81 };

 96 /**
 97  * Returns the appropriate gl constant that identifies the current magnification
 98  * filter applied to this texture
 99  * @param {Object} gl the gl context
100  */
101 vxlTexture.prototype.getMagFilter = function(gl){
102     
103   var tf = vxl.def.texture.filter;  
104   switch(this.mag){
105       case tf.LINEAR: return gl.LINEAR; break;
106       case tf.NEAREST: return gl.NEAREST; break;
107       default: return gl.NEAREST; 
108   }
109 };
110 
111 /**
112  * Returns the appropriate gl constant that identifies the current minification filter
113  * applied to this texture
114  * @param {Object} gl the gl context 
115  */
116 vxlTexture.prototype.getMinFilter = function(gl){
117     var tf = vxl.def.texture.filter;
118     switch(this.min){
119       case tf.LINEAR: return gl.LINEAR; break;
120       case tf.NEAREST: return gl.NEAREST; break;
121       case tf.LINEAR_MIPMAP_LINEAR : return gl.LINEAR_MIPMAP_LINEAR; break;
122       case tf.LINEAR_MIPMAP_NEAREST: return gl.LINEAR_MIPMAP_NEAREST; break;
123       case tf.NEAREST_MIPMAP_LINEAR: return gl.NEAREST_MIPMAP_LINEAR; break;
124       case tf.NEAREST_MIPMAP_NEAREST: return gl.NEAREST_MIPMAP_NEAREST; break;
125       default: return gl.NEAREST; 
126   }
127 };
