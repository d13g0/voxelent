 17 /**
 18  * @constructor
 19  * @class 
 20  * Picking in voxelent is based on colors. The vxlPicker class keeps track of the colors
 21  * in the scene that are used to identify objects and cells. The picker contains a map
 22  * that allows recognizing an object given a color.
 23  * @author Diego Cantor
 24  */
 25 function vxlPicker(){
 26     this._map = {};
 27     
 28     this._hmap = [];
 29     
 30     for (var i=0;i<=256; i+=1){
 31         this._hmap[i] = [];
 32         for(var j=0;j<=256; j+=1){
 33             this._hmap[i][j] = [];
 34             for(var k=0;k<=256; k+=1){
 35                 this._hmap[i][j][k] = null;
 36                 }
 37         }
 38     }    
 39 };
 40 
 41 
 42 /**
 43  * Generates a color that has not been assigned to any object or cell in the scene
 44  */
 45 vxlPicker.prototype._getColor = function(){
 46     
 47     function getN(){
 48         var x =  Math.floor(Math.random()*256);
 49         if (x == 0) 
 50             return getN();
 51         else
 52             return x;
 53     };
 54     
 55  
 56     return [getN(), getN(), getN()];
 57 };
 58   
 59 /**
 60  * 
 61  */
 62 vxlPicker.prototype.color2decimal = function(color){
 63     r = Math.round((color[0]/256)*100)/100;
 64     g = Math.round((color[1]/256)*100)/100;
 65     b = Math.round((color[2]/256)*100)/100;
 66     return [r,g,b];
 67 }  
 68 
 69 
 70 /**
 71  * @param {Object} obj an object that can be either a vxlCell or a vxlActor
 72  */
 73 vxlPicker.prototype.getColorFor = function(obj){
 74     
 75     var uid = obj.UID;
 76     
 77     if (uid == null || uid  == undefined){
 78         alert("vxlPicker.getColor: invalid object");
 79         return;
 80     }
 81     
 82     if(!this._map[uid]){
 83         
 84         var color; 
 85         do{
 86             color = this._getColor();
 87         } while(this._hmap[color[0]][color[1]][color[2]] != null);
 88 
 89         this._map[uid] =  color;
 90         this._hmap[color[0]][color[1]][color[2]] = uid;
 91     }
 92     return this.color2decimal(color);
 93 };
 94 
 95 /**
 96  * Checks if the color passed as a parameter correspond to any UID (object,cell) assigned in the picker
 97  * If so, it returns an object with the results
 98  * If not, it returns null indicating the query was unsuccessful.
 99  * @param {Array} color
100  * 
101  */
102 vxlPicker.prototype.query = function(color){
103     
104     var distance = 100;
105     var closest_uid = undefined;
106     var results = {}
107     var v = 2;
108     
109     if (this._hmap[color[0]][color[1]][color[2]] != null){
110         results.uid = this._hmap[color[0]][color[1]][color[2]];
111         results.color = color;
112         return results;
113     }
114     
115         
116     for (var i=-v;i<=v;i+=1){
117         for (var j=-v;j<=v;j+=1){
118             for (var k=-v;k<=v;k+=1){
119                 if (color[0]+i<0 || color[0]+i>256 || 
120                     color[1]+j<0 || color[1]+j>256 || 
121                     color[2]+k<0 || color[2]+k>256) continue;
122                 var closest_uid = this._hmap[color[0]+i][color[1]+j][color[2]+k];
123                 if (closest_uid != null){
124                     results.uid = closest_uid;
125                     results.color = [color[0]+i,color[1]+j,color[2]+k];
126                     return results;
127                 }
128             }
129         }
130     }
131     return null;
132             
133 
134 };
