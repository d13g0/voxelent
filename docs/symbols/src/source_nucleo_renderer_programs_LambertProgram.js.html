 45     VERTEX_SHADER : [
 46     
 47     "attribute vec3 aVertexPosition;",
 48 	"attribute vec3 aVertexNormal;",
 49 	"attribute vec3 aVertexColor;",
 50     "uniform float uPointSize;",
 51 	"uniform mat4 mModelView;",
 52 	"uniform mat4 mPerspective;",
 53 	"uniform mat4 mModelViewPerspective;",
 54 	"uniform mat4 mNormal;",
 55 	"uniform vec3 uLightDirection;",
 56 	"uniform vec4 uLightAmbient;",  
 57 	"uniform vec4 uLightDiffuse;",
 58 	"uniform vec4 uMaterialDiffuse;",
 59 	"uniform bool uUseShading;",
 60     "uniform bool uUseVertexColors;",
 61     "uniform bool uUseLightTranslation;",
 62 	"varying vec4 vFinalColor;",
 63     
 64     "void main(void) {",

 77 	"		float lambertTerm = max(dot(N,-L),0.4);",
 78 	"		vec4 Ia = uLightAmbient;",
 79 	"		vec4 Id = vFinalColor * uLightDiffuse * lambertTerm;",
 80     "		vFinalColor = Ia + Id;",
 81 	"		vFinalColor.a = uMaterialDiffuse.a;",
 82 	"	}" ,
 83 	"}"].join('\n'),
