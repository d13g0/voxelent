 18 vxl.def.model.boundingBox = new vxlModel();
 19 vxl.def.model.boundingBox.load('bounding box', { "vertices" : [], "wireframe":[0,1,1,2,2,3,3,0,0,4,4,5,5,6,6,7,7,4,1,5,2,6,3,7], "diffuse":[1.0,1.0,1.0,1.0]});
 20 
 21 //vxlBoundingBox IS a vxlActor                                               
 22 vxlBoundingBox.prototype = new vxlActor();
 23 vxlBoundingBox.prototype.constructor = vxlBoundingBox;
 24 
 25 /**
 26  * @class
 27  * @constructor
 28  * @extends vxlActor
 29  */
 30 function vxlBoundingBox() {
 31 	vxlActor.call(this, vxl.def.model.boundingBox);
 32 	this.bb 		= [];
 33     this.mode 		= vxl.def.actor.mode.WIREFRAME;
 34     this.visible 	= false;
 35     this.toy    	= true;
 36     this.shading    = false;
 37 };
 38 
 39 /**
 40 * Sets the bounding box
 41 * @param {Array} b the bounding box. The format of this param should be [x1,y1,z1,x2,y2,z2]
 42 * where x1,y1,z1 correspond to the minimum bounding coordinates and x2,y2,z2 correspond to the
 43 * maximum bounding coordinates
 44 */
 45 vxlBoundingBox.prototype.setBoundingBox = function(b){
 46 	this.bb = [
 47 		b[0], b[1], b[2],
 48 		b[0], b[4], b[2],
 49 		b[3], b[4], b[2],
 50 		b[3], b[1], b[2],
 51 		b[0], b[1], b[5],
 52 		b[0], b[4], b[5],
 53 		b[3], b[4], b[5],
 54 		b[3], b[1], b[5] 
 55 		];
 56         
 57     this.model.vertices = this.bb;
 58 };
